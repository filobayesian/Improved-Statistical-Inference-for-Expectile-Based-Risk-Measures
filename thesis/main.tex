\section{Proof of \texorpdfstring{\Cref{prop:pool-int:A-B-equiv}}{Proposition 3.1}}
\label{app:proofs:A-B-equiv}

We prove the $\sqrt{k}$-asymptotic equivalence of the two pooled
QB intermediate-expectile estimators
$\hatexpectile{\tau_{n}}^{(A)}$ and $\hatexpectile{\tau_{n}}^{(B)}$
introduced in \Cref{sec:pool-int:estimator}, applied to a common,
\emph{deterministic} weight vector $\bm\omega \in \R^{m}$ with
$\bm\omega^{\top}\bm 1 = 1$, the number of samples $m$ being fixed
as $n \to \infty$.

\paragraph{Reduction to a $\psifn$-difference.}
From the definitions \eqref{eq:pool-int:expectile-A} and
\eqref{eq:pool-int:expectile-B},
\begin{align*}
  \hatexpectile{\tau_{n}}^{(A)}(\bm\omega, \bm\omega)
    &\;=\;
    \psifn(\hat\gamma_{n}(\bm\omega))\,
    \hatWei_{n}(1 - \tau_{n} \mid \bm\omega), \\
  \hatexpectile{\tau_{n}}^{(B)}(\bm\omega)
    &\;=\;
    \Big(\prod_{j = 1}^{m}\psifn(\hat\gamma_{j}(k_{j}))^{\omega_{j}}\Big)
    \,\hatWei_{n}(1 - \tau_{n} \mid \bm\omega).
\end{align*}
The Weissman factor is identical in the two estimators, so the
ratio reduces to a function of the Hill estimators only:
\begin{equation}
\label{eq:app:A-B-ratio}
  \log\!\frac{\hatexpectile{\tau_{n}}^{(A)}(\bm\omega, \bm\omega)}
              {\hatexpectile{\tau_{n}}^{(B)}(\bm\omega)}
  \;=\;
  \log\psifn\!\big(\hat\gamma_{n}(\bm\omega)\big)
  \;-\;
  \sum_{j = 1}^{m}\omega_{j}\,
    \log\psifn\!\big(\hat\gamma_{j}(k_{j})\big).
\end{equation}
The proposition therefore reduces to
\begin{equation}
\label{eq:app:reduction-target}
  \sqrt{k}\,\Big[
    \log\psifn(\hat\gamma_{n}(\bm\omega))
    - \sum_{j}\omega_{j}\log\psifn(\hat\gamma_{j}(k_{j}))
  \Big] \;\toprob\; 0.
\end{equation}

\paragraph{Good event supporting the Taylor expansion.}
The map $g \mapsto \log\psifn(g) = -g\log(g^{-1} - 1)$ is defined
on the open interval $(0, 1)$ and singular at its endpoints, so the
Taylor expansion below requires $\hat\gamma_{j}(k_{j})$ and
$\hat\gamma_{n}(\bm\omega)$ to lie in $(0, 1)$. By
\Cref{thm:bg:hill}, $\hat\gamma_{j}(k_{j}) \toprob \gamma$ for every
$j$, and $\gamma \in (0, 1/2)$ by \eqref{eq:pool-int:moment}; hence
\[
  \mathbb{P}\!\big(\,\hat\gamma_{j}(k_{j}) \in (0, 1)
    \text{ for every } j = 1, \ldots, m\,\big) \;\to\; 1.
\]
The pooled estimator inherits the same property:
$\hat\gamma_{n}(\bm\omega)
= \sum_{j} \omega_{j}\,\hat\gamma_{j}(k_{j}) \toprob \gamma$ (a
finite weighted sum of consistent terms with deterministic weights
summing to one), so
$\mathbb{P}(\hat\gamma_{n}(\bm\omega) \in (0, 1)) \to 1$. Define
the event
\[
  E_{n} \;\coloneqq\; \big\{\hat\gamma_{1}(k_{1}), \ldots,
    \hat\gamma_{m}(k_{m}),\, \hat\gamma_{n}(\bm\omega) \in (0, 1)\big\},
  \qquad \mathbb{P}(E_{n}) \to 1.
\]
It suffices to establish the claimed $O_{p}(k^{-1})$ bound on
$E_{n}$: for any $\epsilon > 0$ and $M > 0$,
$\mathbb{P}(|X_{n}| > M a_{n}) \le \mathbb{P}(|X_{n}|\mathbf{1}_{E_{n}}
> M a_{n}) + \mathbb{P}(E_{n}^{c})$, and the second term tends to
zero, so modifying a sequence on an event of vanishing probability
does not affect convergence in probability or $O_{p}(\cdot)$ bounds.
Throughout the remainder of the proof we therefore work on $E_{n}$
without further comment.

\paragraph{Taylor expansion with Lagrange remainder.}
The map $g \mapsto \log\psifn(g) = -g\log(g^{-1} - 1)$ is
$C^{\infty}$ on the open interval $(0, 1)$, with first derivative
$m(g) = (1 - g)^{-1} - \log(g^{-1} - 1)$ (the function introduced
in \eqref{eq:pool-int:m-function}) and second derivative
\[
  m'(g)
  \;=\;
  \frac{1}{(1 - g)^{2}}
  + \frac{1}{1 - g}
  + \frac{1}{g}
  \;>\; 0,
  \qquad g \in (0, 1).
\]
By Taylor's theorem with Lagrange remainder, for each $j \in
\{1, \ldots, m\}$ there exists a random $\tilde\gamma_{j}$ lying
between $\hat\gamma_{j}(k_{j})$ and $\gamma$ such that
\begin{equation}
\label{eq:app:taylor-j}
  \log\psifn(\hat\gamma_{j}(k_{j}))
  - \log\psifn(\gamma)
  \;=\;
  m(\gamma)\,\big(\hat\gamma_{j}(k_{j}) - \gamma\big)
  + \tfrac{1}{2}\,m'(\tilde\gamma_{j})\,
    \big(\hat\gamma_{j}(k_{j}) - \gamma\big)^{2},
\end{equation}
and similarly there exists $\tilde\gamma_{\star}$ between
$\hat\gamma_{n}(\bm\omega)$ and $\gamma$ with
\begin{equation}
\label{eq:app:taylor-pool}
  \log\psifn(\hat\gamma_{n}(\bm\omega))
  - \log\psifn(\gamma)
  \;=\;
  m(\gamma)\,\big(\hat\gamma_{n}(\bm\omega) - \gamma\big)
  + \tfrac{1}{2}\,m'(\tilde\gamma_{\star})\,
    \big(\hat\gamma_{n}(\bm\omega) - \gamma\big)^{2}.
\end{equation}
Subtract the $\bm\omega$-weighted sum of \eqref{eq:app:taylor-j}
across $j$ from \eqref{eq:app:taylor-pool}. The constant terms
$\log\psifn(\gamma)$ cancel because $\sum_{j}\omega_{j} = 1$, and
the linear terms cancel because the pooled Hill estimator is the
weighted linear combination
$\hat\gamma_{n}(\bm\omega) = \sum_{j}\omega_{j}\hat\gamma_{j}(k_{j})$,
which implies $\sum_{j}\omega_{j}(\hat\gamma_{j} - \gamma) =
\hat\gamma_{n}(\bm\omega) - \gamma$. We obtain
\begin{equation}
\label{eq:app:A-B-master}
  \log\psifn(\hat\gamma_{n}(\bm\omega))
  - \sum_{j}\omega_{j}\log\psifn(\hat\gamma_{j}(k_{j}))
  \;=\;
  \tfrac{1}{2}\,m'(\tilde\gamma_{\star})\,
    \big(\hat\gamma_{n}(\bm\omega) - \gamma\big)^{2}
  - \tfrac{1}{2}\sum_{j}\omega_{j}\,m'(\tilde\gamma_{j})\,
    \big(\hat\gamma_{j}(k_{j}) - \gamma\big)^{2}.
\end{equation}

\paragraph{Order of the remainder terms.}
By \Cref{thm:bg:hill},
$\sqrt{k_{j}}(\hat\gamma_{j}(k_{j}) - \gamma) \todist
\mathcal N(\lambda_{j}/(1 - \rho), \gamma^{2})$ for every $j$,
hence $\hat\gamma_{j}(k_{j}) - \gamma = O_{p}(k_{j}^{-1/2})$. The
proportionality condition \eqref{eq:bg:pool-rates}, $k_{1}/k_{j}
\to c_{j} \in (0, \infty)$, gives $k_{j} \asymp k$ uniformly in
$j$, so
\begin{equation}
\label{eq:app:Op-bound-per-sample}
  \hat\gamma_{j}(k_{j}) - \gamma \;=\; O_{p}(k^{-1/2}),
  \qquad
  \big(\hat\gamma_{j}(k_{j}) - \gamma\big)^{2} \;=\; O_{p}(k^{-1}),
  \qquad j = 1, \ldots, m.
\end{equation}
The pooled Hill estimator inherits the rate as a finite weighted
combination of $\sqrt{k}$-consistent terms:
\begin{equation}
\label{eq:app:Op-bound-pool}
  \hat\gamma_{n}(\bm\omega) - \gamma \;=\; O_{p}(k^{-1/2}),
  \qquad
  \big(\hat\gamma_{n}(\bm\omega) - \gamma\big)^{2}
    \;=\; O_{p}(k^{-1}).
\end{equation}
For the Lagrange-remainder pre-factors, the consistency
$\hat\gamma_{j}(k_{j}) \toprob \gamma$ and the squeeze
$\tilde\gamma_{j}$ between $\hat\gamma_{j}$ and $\gamma$ give
$\tilde\gamma_{j} \toprob \gamma$; continuity of $m'$ on
$(0, 1)$ then yields, by the continuous mapping theorem,
$m'(\tilde\gamma_{j}) \toprob m'(\gamma) = O_{p}(1)$. The same
argument gives $m'(\tilde\gamma_{\star}) \toprob m'(\gamma)$.

Inserting these orders into \eqref{eq:app:A-B-master}: each summand
on the right-hand side is the product of an $O_{p}(1)$ factor
($m'(\tilde\gamma_{\cdot})$) and an $O_{p}(k^{-1})$ factor (a
squared deviation), so each is $O_{p}(k^{-1})$. Since the sum on
the right has $m + 1$ terms and $m$ is fixed, the right-hand side
of \eqref{eq:app:A-B-master} is itself $O_{p}(k^{-1})$.

\paragraph{Conclusion.}
The right-hand side of \eqref{eq:app:A-B-master} is a finite sum of
$m + 1$ terms, each a product of an $O_{p}(1)$ Lagrange factor and
an $O_{p}(k^{-1})$ squared deviation. With $m$ fixed,
\[
  \log\psifn(\hat\gamma_{n}(\bm\omega))
  - \sum_{j}\omega_{j}\log\psifn(\hat\gamma_{j}(k_{j}))
  \;=\; O_{p}(k^{-1}).
\]
Combining this with the reduction \eqref{eq:app:A-B-ratio} gives
\[
  \log\!\frac{\hatexpectile{\tau_{n}}^{(A)}(\bm\omega, \bm\omega)}
              {\hatexpectile{\tau_{n}}^{(B)}(\bm\omega)}
  \;=\; O_{p}(k^{-1}),
\]
hence
\[
  \sqrt{k}\,
  \log\!\frac{\hatexpectile{\tau_{n}}^{(A)}(\bm\omega, \bm\omega)}
              {\hatexpectile{\tau_{n}}^{(B)}(\bm\omega)}
  \;=\; O_{p}(k^{-1/2}) \;=\; o_{p}(1),
\]
which proves the convergence in probability claimed by
\Cref{prop:pool-int:A-B-equiv}. Since $e^{x} - 1 \sim x$ as
$x \to 0$ and our log-ratio is $O_{p}(k^{-1})$, the continuous
mapping theorem applied to $\exp$ further yields
\[
  \frac{\hatexpectile{\tau_{n}}^{(A)}(\bm\omega, \bm\omega)}
       {\hatexpectile{\tau_{n}}^{(B)}(\bm\omega)} - 1
  \;=\; O_{p}(k^{-1}).
\]
The $k^{-1}$ rate is one full order \emph{stronger} than the
$o_{p}(k^{-1/2})$ statement that the $\sqrt{k}$-CLT
\Cref{thm:pool-int:main} requires: the two constructions agree
asymptotically not only at the CLT scale but at a strictly finer
scale, so the two estimators are interchangeable to leading order
in every quantity downstream of the joint Gaussian limit. \qed

\section{Proof of \texorpdfstring{\Cref{prop:pool-int:joint-clt}}{Proposition 3.2}}
\label{app:proofs:joint-clt}

We establish the joint $\sqrt{k}$-Gaussian limit
\eqref{eq:pool-int:joint-pooled-clt} of the pair
$\big(\hat\gamma_{n}(\bm\omega_{\gamma}),\,
\log\hatWei_{n}(1 - \tau_{n} \mid \bm\omega_{q})\big)$ under the
standing assumptions of \Cref{sec:pool-int:setup}: $m$ independent
samples sharing the common marginal $F$ and the common tail index
$\gamma \in (0, 1/2)$, the proportionality conditions
\eqref{eq:bg:pool-rates}, the second-order condition
$\mathcal C_{2}(\gamma, \rho, A)$ with the per-sample bias rate
$\sqrt{k_{j}}\,A(n_{j}/k_{j}) \to \lambda_{j} \in \R$, an intermediate
target level $\tau_{n} \to 1$ with $n(1 - \tau_{n}) \to \infty$, and
the chapter's log-rate convergence $L_{j}(\tau_{n}) \to L_{j}^{\star}
\in \R$ for every $j$. The number of samples $m$ is fixed.

The proof composes four steps:
(i) the within-sample joint CLT for the Hill estimator and the
intermediate log-order statistic;
(ii) the linear lift of (i) to the (Hill, log-Weissman) pair via the
exact decomposition \eqref{eq:pool-int:weissman-decomp};
(iii) cross-sample independence, which renders the $2m$-vector of all
per-sample inputs jointly Gaussian with block-diagonal covariance on
the common $\sqrt{k}$ scale;
(iv) the linear pooling map, which collapses the $2m$-vector onto the
$2$-vector \eqref{eq:pool-int:joint-pooled-clt} and yields the
covariance \eqref{eq:pool-int:Sigma}.

\paragraph{Step (i): within-sample joint CLT for Hill and intermediate log-order statistic.}
Fix $j \in \{1, \ldots, m\}$ and write $\tau_{n_{j}} \coloneqq
1 - k_{j}/n_{j}$ for the sample-$j$ intermediate level. The joint
$\sqrt{k_{j}}$-asymptotics of the Hill estimator and its companion
intermediate log-order statistic is a standard
tail-empirical-process result
\parencite[Theorem~2.4.8 and \S5.1]{deHaanFerreira2006}; in the form
we use it is the input to the iid QB plug-in CLT of
\textcite{DaouiaGirardStupfler2018} reproduced as
\Cref{thm:bg:qb-iid}. Under $\mathcal C_{2}(\gamma, \rho, A)$ with
$k_{j} \to \infty$, $k_{j}/n_{j} \to 0$, and
$\sqrt{k_{j}}\,A(n_{j}/k_{j}) \to \lambda_{j} \in \R$,
\begin{equation}
\label{eq:app:within-input}
  \sqrt{k_{j}}\!
    \begin{pmatrix}
      \hat\gamma_{j}(k_{j}) - \gamma \\[2pt]
      \log X_{n_{j} - k_{j} : n_{j},\,j}
        - \log \Quant_{X}(\tau_{n_{j}})
    \end{pmatrix}
  \;\todist\;
  \mathcal N\!\left(
    \begin{pmatrix} b_{j}^{H} \\[2pt] b_{j}^{Q} \end{pmatrix},\;
    \gamma^{2}\, \bm I_{2}\right),
\end{equation}
where the bias parameters
$b_{j}^{H} = \lambda_{j}/(1 - \rho)$ and
$b_{j}^{Q} = b_{j}^{Q}(\lambda_{j}, \rho)$ are determined by the
second-order condition, and $\bm I_{2}$ is the $2 \times 2$ identity
matrix. The load-bearing structural fact in
\eqref{eq:app:within-input} is the \emph{zero off-diagonal entry}:
the Hill estimator is a smooth functional of the centred log-spacings
above $X_{n_{j} - k_{j} : n_{j},\,j}$, while the
intermediate-threshold fluctuation is the centred $(k_{j} + 1)$-th
upper log-order statistic, and the two decouple in the
tail-empirical-process limit with common variance $\gamma^{2}$. The
diagonal form $\gamma^{2}\bm I_{2}$ is therefore essential for the
simple $\Vc$-and-$\mathrm{diag}(\bm L^{\star})$ covariance structure
of \eqref{eq:pool-int:Sigma} used in the rest of the chapter.

\paragraph{Step (ii): linear lift to (Hill, log-Weissman).}
The Weissman decomposition \eqref{eq:pool-int:weissman-decomp},
exact for every $n$, reads
\begin{equation}
\label{eq:app:wei-decomp}
  \log\hat q^{\star}_{j}(1 - \tau_{n} \mid k_{j})
    \;-\; \log\Quant_{X}(\tau_{n})
  \;=\;
  L_{j}(\tau_{n})\,\big(\hat\gamma_{j}(k_{j}) - \gamma\big)
  \;+\;
  \big(\log X_{n_{j} - k_{j} : n_{j},\,j}
       - \log\Quant_{X}(\tau_{n_{j}})\big)
  \;+\; r_{j}(\tau_{n}),
\end{equation}
where
$r_{j}(\tau_{n}) \coloneqq
\gamma\,L_{j}(\tau_{n}) + \log\Quant_{X}(\tau_{n_{j}})
- \log\Quant_{X}(\tau_{n})$
is a deterministic remainder. Identity \eqref{eq:app:wei-decomp}
follows by inserting the definition
$\hat q^{\star}_{j}(1 - \tau_{n} \mid k_{j})
= (k_{j}/(n_{j}(1 - \tau_{n})))^{\hat\gamma_{j}(k_{j})}
X_{n_{j} - k_{j} : n_{j},\,j}$ and adding and subtracting
$\gamma\,L_{j}(\tau_{n})$ and $\log\Quant_{X}(\tau_{n_{j}})$.

The remainder $r_{j}(\tau_{n})$ measures the deterministic gap
between the population log-quantile at the chapter level $\tau_{n}$
and its first-order regular-variation approximation pivoted at
$\tau_{n_{j}}$. By the second-order condition $\mathcal C_{2}(\gamma,
\rho, A)$ applied to $U = (1/\overline F)^{\leftarrow}$ at base point
$t = n_{j}/k_{j}$ and ratio $y = k_{j}/(n_{j}(1 - \tau_{n}))$, one has
$\log U(t\,y) - \log U(t) - \gamma\log y = A(t)\,\Psi_{\rho}(y) +
o(A(t))$ uniformly on compact $y$-sets, with
$\Psi_{\rho}(y) = (y^{\rho} - 1)/\rho$. Therefore
\begin{equation}
\label{eq:app:rj-bias}
  r_{j}(\tau_{n})
  \;=\;
  -A(n_{j}/k_{j})\,
   \Psi_{\rho}\!\big(k_{j}/(n_{j}(1 - \tau_{n}))\big)
  \;+\; o\big(A(n_{j}/k_{j})\big),
\end{equation}
and the per-sample bias rate $\sqrt{k_{j}}\,A(n_{j}/k_{j}) \to
\lambda_{j}$ together with $L_{j}(\tau_{n}) \to L_{j}^{\star} \in \R$
yields $\sqrt{k_{j}}\,r_{j}(\tau_{n}) \to -\lambda_{j}\,
\Psi_{\rho}(e^{L_{j}^{\star}}) =: b_{j}^{R}$, a finite deterministic
shift. Since $r_{j}(\tau_{n})$ is deterministic, it shifts only the
limiting mean and does not enter the covariance: the asymptotic
covariance below is determined entirely by the two random inputs
$\hat\gamma_{j}(k_{j}) - \gamma$ and
$\log X_{n_{j} - k_{j} : n_{j},\,j} - \log\Quant_{X}(\tau_{n_{j}})$
of \eqref{eq:app:within-input}.

Write \eqref{eq:app:wei-decomp} in matrix form,
\[
  \sqrt{k_{j}}\!
    \begin{pmatrix}
      \hat\gamma_{j}(k_{j}) - \gamma \\[2pt]
      \log\hat q^{\star}_{j}(1 - \tau_{n} \mid k_{j})
        - \log\Quant_{X}(\tau_{n})
    \end{pmatrix}
  \;=\;
  \underbrace{\!\begin{pmatrix} 1 & 0 \\ L_{j}(\tau_{n}) & 1
              \end{pmatrix}\!}_{=:\,\bm M_{j}(\tau_{n})}\,
  \sqrt{k_{j}}\!
    \begin{pmatrix}
      \hat\gamma_{j}(k_{j}) - \gamma \\[2pt]
      \log X_{n_{j} - k_{j} : n_{j},\,j}
        - \log\Quant_{X}(\tau_{n_{j}})
    \end{pmatrix}
  \;+\;
  \begin{pmatrix} 0 \\ \sqrt{k_{j}}\,r_{j}(\tau_{n}) \end{pmatrix}.
\]
The deterministic matrix $\bm M_{j}(\tau_{n}) \to \bm M_{j}^{\star}
\coloneqq \big(1, 0;\, L_{j}^{\star}, 1\big)$ entrywise, and the
deterministic shift converges to $(0, b_{j}^{R})^{\top}$. The
continuous mapping theorem and Slutsky combine \eqref{eq:app:within-input}
with these deterministic convergences to deliver
\begin{equation}
\label{eq:app:within-output}
  \sqrt{k_{j}}\!
    \begin{pmatrix}
      \hat\gamma_{j}(k_{j}) - \gamma \\[2pt]
      \log\hat q^{\star}_{j}(1 - \tau_{n} \mid k_{j})
        - \log\Quant_{X}(\tau_{n})
    \end{pmatrix}
  \;\todist\;
  \mathcal N\!\left(\bm\mu_{j}^{\star},\;
                   \gamma^{2}\!\begin{pmatrix}
                     1 & L_{j}^{\star} \\
                     L_{j}^{\star} & 1 + (L_{j}^{\star})^{2}
                   \end{pmatrix}\right),
\end{equation}
with mean
$\bm\mu_{j}^{\star} = \bm M_{j}^{\star}(b_{j}^{H}, b_{j}^{Q})^{\top}
+ (0, b_{j}^{R})^{\top}
= \big(b_{j}^{H},\, L_{j}^{\star}\, b_{j}^{H} + b_{j}^{Q}
       + b_{j}^{R}\big)^{\top}$.
The covariance is
$\bm M_{j}^{\star}\,(\gamma^{2}\bm I_{2})\,(\bm M_{j}^{\star})^{\top}
= \gamma^{2}\big(1, L_{j}^{\star};\, L_{j}^{\star}, 1+(L_{j}^{\star})^{2}\big)$,
which is precisely \eqref{eq:pool-int:per-sample-clt}.

\paragraph{Step (iii): cross-sample independence and rescaling to $\sqrt{k}$.}
The $m$ samples are independent, so the $2m$ random variables
$\big(\hat\gamma_{j}(k_{j}) - \gamma,\,
\log\hat q^{\star}_{j}(1 - \tau_{n} \mid k_{j}) -
\log\Quant_{X}(\tau_{n})\big)_{j = 1}^{m}$ are organised as $m$
independent $2$-blocks. Stacking them into the column vector
\[
  \bm Z_{n}
  \;\coloneqq\;
  \big(\hat\gamma_{1} - \gamma,\, \ldots,\,
       \hat\gamma_{m} - \gamma,\;
       \log\hat q^{\star}_{1} - \log\Quant_{X}(\tau_{n}),\, \ldots,\,
       \log\hat q^{\star}_{m} - \log\Quant_{X}(\tau_{n})\big)^{\!\top}
  \in \R^{2m}
\]
(Hill components stacked first, log-Weissman components second), the
joint CLT of $\sqrt{k_{j}}\bm Z_{n}^{(j)}$ from
\eqref{eq:app:within-output} aggregates across $j$ under independence
to a joint $2m$-dimensional Gaussian limit. To express the limit on
the common $\sqrt{k}$ scale, multiply each per-sample $\sqrt{k_{j}}$
input by the deterministic factor $\sqrt{k/k_{j}}$, which converges
to a finite limit:
\[
  \frac{k}{k_{j}}
  \;=\;
  \frac{\sum_{i = 1}^{m} k_{i}}{k_{j}}
  \;=\;
  \sum_{i = 1}^{m} \frac{k_{1}/k_{j}}{k_{1}/k_{i}}
  \;\longrightarrow\;
  c_{j}\sum_{i = 1}^{m} c_{i}^{-1},
\]
using \eqref{eq:bg:pool-rates}. The per-sample within-block
covariance \eqref{eq:app:within-output} scales by the factor
$k/k_{j}$, so its diagonal entry
$(\sqrt{k}/\sqrt{k_{j}})^{2}\cdot \gamma^{2}
\to \gamma^{2}\,c_{j}\,\sum_{i}c_{i}^{-1}
= (\Vc)_{j,j}$, the diagonal entry of the diagonal pooled-Hill
covariance \eqref{eq:bg:V_c-diagonal} for the
independent-samples specialisation. We therefore obtain
\begin{equation}
\label{eq:app:Z-joint-CLT}
  \sqrt{k}\,\bm Z_{n}
  \;\todist\;
  \mathcal N\!\big(\bm\mu_{Z},\, \bm\Sigma_{Z}\big),
\end{equation}
where the $2m \times 2m$ covariance $\bm\Sigma_{Z}$ has the
block-structured form
\begin{equation}
\label{eq:app:Sigma-Z}
  \bm\Sigma_{Z}
  \;=\;
  \begin{pmatrix}
    \Vc & \mathrm{diag}(\bm L^{\star})\,\Vc \\[2pt]
    \mathrm{diag}(\bm L^{\star})\,\Vc
      & \Vc + \mathrm{diag}(\bm L^{\star})^{2}\,\Vc
  \end{pmatrix},
\end{equation}
expressed in $m \times m$ blocks. Each block is diagonal: the
upper-left block is $\Vc$ itself (with diagonal entries
$(\Vc)_{j,j} = \gamma^{2}\,c_{j}\sum_{i}c_{i}^{-1}$); the
off-diagonal blocks have $j$th diagonal entry
$L_{j}^{\star}(\Vc)_{j,j}$; the lower-right block has $j$th diagonal
entry $(1 + (L_{j}^{\star})^{2})(\Vc)_{j,j}$. All non-diagonal
entries within each block vanish because of cross-sample
independence. The mean $\bm\mu_{Z}$ is the corresponding linear
functional of the per-sample bias parameters $b_{j}^{H}, b_{j}^{Q},
b_{j}^{R}$, rescaled to the $\sqrt{k}$ scale: explicitly,
$(\bm\mu_{Z})_{j} = \sqrt{c_{j}\sum_{i}c_{i}^{-1}}\, b_{j}^{H}
= (\Bc)_{j}$ for $j = 1, \ldots, m$, and
$(\bm\mu_{Z})_{m + j} = \sqrt{c_{j}\sum_{i}c_{i}^{-1}}\,
\big(L_{j}^{\star}\, b_{j}^{H} + b_{j}^{Q} + b_{j}^{R}\big)
=: (\Bc^{\star})_{j}$. The first half of $\bm\mu_{Z}$ is
exactly the pooled-Hill bias vector $\Bc$ of
\Cref{thm:bg:pool-hill}; the second half is the
pooled-Weissman bias vector $\Bc^{\star}$ specialised to
the intermediate-level target.

\paragraph{Step (iv): linear pooling map.}
By definitions \eqref{eq:pool-int:hill-pool} and
\eqref{eq:pool-int:weissman-pool},
\[
  \hat\gamma_{n}(\bm\omega_{\gamma}) - \gamma
  \;=\;
  \sum_{j = 1}^{m}\omega_{\gamma, j}\,(\hat\gamma_{j}(k_{j}) - \gamma),
  \qquad
  \log\hatWei_{n}(1 - \tau_{n} \mid \bm\omega_{q})
    - \log\Quant_{X}(\tau_{n})
  \;=\;
  \sum_{j = 1}^{m}\omega_{q, j}\,
    \big(\log\hat q^{\star}_{j} - \log\Quant_{X}(\tau_{n})\big),
\]
where the second identity uses $\bm\omega_{q}^{\top}\bm 1 = 1$ so
that subtracting the common term $\log\Quant_{X}(\tau_{n})$ commutes
with the weighted sum. Both pooled coordinates are linear in $\bm
Z_{n}$, and the relation is exactly
\[
  \begin{pmatrix}
    \hat\gamma_{n}(\bm\omega_{\gamma}) - \gamma \\[2pt]
    \log\hatWei_{n}(1 - \tau_{n} \mid \bm\omega_{q})
      - \log\Quant_{X}(\tau_{n})
  \end{pmatrix}
  \;=\;
  \bm A\,\bm Z_{n},
  \qquad
  \bm A
  \;\coloneqq\;
  \begin{pmatrix}
    \bm\omega_{\gamma}^{\top} & \bm 0_{m}^{\top} \\
    \bm 0_{m}^{\top} & \bm\omega_{q}^{\top}
  \end{pmatrix}
  \;\in \R^{2 \times 2m}.
\]
Applying $\bm A$ to \eqref{eq:app:Z-joint-CLT},
\[
  \sqrt{k}\,\bm A\,\bm Z_{n}
  \;\todist\;
  \mathcal N\!\big(\bm A\,\bm\mu_{Z},\,
                    \bm A\,\bm\Sigma_{Z}\,\bm A^{\top}\big).
\]
The mean follows from the block structure of $\bm\mu_{Z}$ identified
in Step (iii):
$\bm A\,\bm\mu_{Z}
= \big(\bm\omega_{\gamma}^{\top}\Bc,\;
       \bm\omega_{q}^{\top}\Bc^{\star}\big)^{\top}$,
which is the chapter's $\bm\mu(\bm\omega_{\gamma}, \bm\omega_{q})$.

For the covariance, partition $\bm A$ along the same Hill /
log-Weissman split as $\bm\Sigma_{Z}$ and compute block-by-block,
using \eqref{eq:app:Sigma-Z}:
\begin{align*}
  \bm A\,\bm\Sigma_{Z}\,\bm A^{\top}\big|_{1,1}
  &\;=\;
  \bm\omega_{\gamma}^{\top}\,\Vc\,\bm\omega_{\gamma},
  \\
  \bm A\,\bm\Sigma_{Z}\,\bm A^{\top}\big|_{1,2}
  &\;=\;
  \bm\omega_{\gamma}^{\top}\,
    \mathrm{diag}(\bm L^{\star})\,\Vc\,\bm\omega_{q},
  \\
  \bm A\,\bm\Sigma_{Z}\,\bm A^{\top}\big|_{2,2}
  &\;=\;
  \bm\omega_{q}^{\top}\,
    \big(\Vc + \mathrm{diag}(\bm L^{\star})^{2}\,\Vc\big)\,\bm\omega_{q},
\end{align*}
with the $(2, 1)$ block equal to the $(1, 2)$ block by symmetry of
$\bm\Sigma_{Z}$. The diagonality of $\Vc$ (under independent
sampling, \eqref{eq:bg:V_c-diagonal}) ensures that $\Vc$ and
$\mathrm{diag}(\bm L^{\star})$ commute, so the off-diagonal block can
equivalently be written
$\bm\omega_{\gamma}^{\top}\Vc\,\mathrm{diag}(\bm L^{\star})\,\bm\omega_{q}$
and is the entrywise sum
$\sum_{j} \omega_{\gamma, j}\,\omega_{q, j}\,L_{j}^{\star}\,
(\Vc)_{j, j}$. This recovers the covariance
\eqref{eq:pool-int:Sigma} of the proposition.

Combining the two displays, the pooled $2$-vector converges in
distribution to a Gaussian with mean
$\bm\mu(\bm\omega_{\gamma}, \bm\omega_{q})$ and covariance
$\bm\Sigma(\bm\omega_{\gamma}, \bm\omega_{q})$, which is precisely
\eqref{eq:pool-int:joint-pooled-clt}. \qed

\section{Proof of \texorpdfstring{\Cref{thm:pool-int:main}}{Theorem 3.3}}
\label{app:proofs:main-clt}

We prove the pooled QB intermediate-expectile CLT
\eqref{eq:pool-int:main-clt} under the standing assumptions of
\Cref{sec:pool-int:setup}: $m$ independent samples with common
marginal $F$, common tail index $\gamma \in (0, 1/2)$, the
proportionality conditions \eqref{eq:bg:pool-rates}, the second-order
condition $\mathcal C_{2}(\gamma, \rho, A)$ with per-sample bias rate
$\sqrt{k_{j}}\,A(n_{j}/k_{j}) \to \lambda_{j} \in \R$, an intermediate
target $\tau_{n} \to 1$ with $n(1 - \tau_{n}) \to \infty$, and the
log-rate convergence $L_{j}(\tau_{n}) \to L_{j}^{\star} \in \R$ for
every $j$. The number of samples $m$ is fixed. We work for arbitrary
deterministic weight vectors $\bm\omega_{\gamma}, \bm\omega_{q} \in
\R^{m}$ satisfying $\bm\omega_{\gamma}^{\top}\bm 1 =
\bm\omega_{q}^{\top}\bm 1 = 1$; the extension to estimated weights is
collected in \Cref{cor:pool-int:CI}.

The proof composes four steps:
(i) a deterministic identity on the log scale that decomposes the
expectile log-ratio into a random Hill increment, a random log-Weissman
increment, and a deterministic expectile--quantile gap;
(ii) a Taylor linearisation of $\log\psifn$ at $\gamma$ that converts
the Hill increment into a linear functional of $\hat\gamma_{n} -
\gamma$ up to a $\sqrt{k}$-negligible remainder;
(iii) the second-order expectile--quantile expansion that controls the
deterministic gap;
(iv) the multivariate delta method applied to
\Cref{prop:pool-int:joint-clt} to read off the asymptotic mean and
covariance, followed by the continuous-mapping theorem applied to
$\exp$.

\paragraph{Step (i): the deterministic log-ratio decomposition.}
By the QB factorisation \eqref{eq:pool-int:expectile-pool},
\[
  \log\hatexpectile{\tau_{n}}^{\mathrm{pool}}\!
       (\bm\omega_{\gamma}, \bm\omega_{q})
  \;=\;
  \log\psifn\!\big(\hat\gamma_{n}(\bm\omega_{\gamma})\big)
  \;+\;
  \log\hatWei_{n}(1 - \tau_{n} \mid \bm\omega_{q}).
\]
Subtract $\log\expectile{\tau_{n}}$ from both sides and insert
$\pm\,\log\psifn(\gamma)\,\pm\,\log\Quant_{X}(\tau_{n})$:
\begin{equation}
\label{eq:app:main:log-decomp}
\begin{aligned}
  \log\hatexpectile{\tau_{n}}^{\mathrm{pool}}
  - \log\expectile{\tau_{n}}
  \;=\;
  &\underbrace{\big[\log\psifn(\hat\gamma_{n}(\bm\omega_{\gamma}))
              - \log\psifn(\gamma)\big]}_{(\mathrm{H})}
  \\[-2pt]
  + \;&\underbrace{\big[\log\hatWei_{n}(1 - \tau_{n} \mid
                                       \bm\omega_{q})
              - \log\Quant_{X}(\tau_{n})\big]}_{(\mathrm{W})}
  \\[-2pt]
  + \;&\underbrace{\big[\log\psifn(\gamma)
              + \log\Quant_{X}(\tau_{n})
              - \log\expectile{\tau_{n}}\big]}_{(\mathrm{G})}.
\end{aligned}
\end{equation}
The identity \eqref{eq:app:main:log-decomp} is exact for every $n$.
Terms $(\mathrm{H})$ and $(\mathrm{W})$ are random and centred at zero
by \Cref{prop:pool-int:joint-clt}; term $(\mathrm{G})$ is deterministic
and depends on the target level $\tau_{n}$ only through population
quantities. The sign in $(\mathrm{G})$ is chosen so that the rescaled
$\sqrt{k}\,(\mathrm{G})$ converges directly to $b^{\xi/\Quant}$ in
\eqref{eq:pool-int:b-xi-Q-def}, eliminating the bookkeeping of an
intermediate minus sign in Step (iii).

\paragraph{Step (ii): Taylor linearisation of $\log\psifn$.}
The argument is identical in spirit to that of
\Cref{app:proofs:A-B-equiv}. The map $g \mapsto \log\psifn(g) =
-g\log(g^{-1} - 1)$ is $C^{\infty}$ on $(0, 1)$ with first derivative
$m(g)$ defined in \eqref{eq:pool-int:m-function} and second derivative
\[
  m'(g)
  \;=\;
  \frac{1}{(1 - g)^{2}}
  + \frac{1}{1 - g}
  + \frac{1}{g}
  \;>\; 0,
  \qquad g \in (0, 1).
\]
By \Cref{thm:bg:hill} and \eqref{eq:pool-int:moment},
$\hat\gamma_{n}(\bm\omega_{\gamma}) \toprob \gamma \in (0, 1/2)
\subset (0, 1)$, so the event
$E_{n} \coloneqq \{\hat\gamma_{n}(\bm\omega_{\gamma}) \in (0, 1)\}$
satisfies $\mathbb{P}(E_{n}) \to 1$ and we may work on it without
affecting convergence in distribution. On $E_{n}$, Taylor's theorem
with Lagrange remainder yields a random $\tilde\gamma_{\star}$ between
$\hat\gamma_{n}(\bm\omega_{\gamma})$ and $\gamma$ such that
\begin{equation}
\label{eq:app:main:psi-taylor}
  \log\psifn(\hat\gamma_{n}(\bm\omega_{\gamma}))
  - \log\psifn(\gamma)
  \;=\;
  m(\gamma)\,
    \big(\hat\gamma_{n}(\bm\omega_{\gamma}) - \gamma\big)
  + \tfrac{1}{2}\,m'(\tilde\gamma_{\star})\,
    \big(\hat\gamma_{n}(\bm\omega_{\gamma}) - \gamma\big)^{2}.
\end{equation}
\Cref{prop:pool-int:joint-clt} gives $\hat\gamma_{n}(\bm\omega_{\gamma})
- \gamma = O_{p}(k^{-1/2})$, hence $(\hat\gamma_{n}
- \gamma)^{2} = O_{p}(k^{-1})$. The squeeze $\tilde\gamma_{\star}
\toprob \gamma$ together with continuity of $m'$ on $(0, 1)$ gives
$m'(\tilde\gamma_{\star}) \toprob m'(\gamma) = O_{p}(1)$. The
Lagrange remainder in \eqref{eq:app:main:psi-taylor} is therefore
$O_{p}(k^{-1}) = o_{p}(k^{-1/2})$, so
\begin{equation}
\label{eq:app:main:H-linearised}
  \sqrt{k}\,(\mathrm{H})
  \;=\;
  m(\gamma)\,\sqrt{k}\,\big(\hat\gamma_{n}(\bm\omega_{\gamma})
                                 - \gamma\big)
  \;+\; o_{p}(1).
\end{equation}

\paragraph{Step (iii): the deterministic expectile--quantile gap.}
The first-order equivalence
$\expectile{\tau}/\Quant_{X}(\tau) \to \psifn(\gamma)$ of
\Cref{prop:bg:expectile-quantile} sharpens, under the second-order
condition $\mathcal C_{2}(\gamma, \rho, A)$ with the moment bound
$\E |X_{-}|^{2 + \delta} < \infty$ of \eqref{eq:pool-int:moment}, to
the second-order expansion
\begin{equation}
\label{eq:app:main:xi-Q-expansion}
  \log\expectile{\tau}
  - \log\psifn(\gamma)
  - \log\Quant_{X}(\tau)
  \;=\;
  c(\gamma, \rho)\, A\!\big((1 - \tau)^{-1}\big)
  \;+\; o\!\big(A((1 - \tau)^{-1})\big),
  \qquad \tau \uparrow 1,
\end{equation}
established in
\textcite[Theorem~2, eq.~(8)]{DaouiaGirardStupfler2018} for the
expectile--quantile ratio (the additive log form
\eqref{eq:app:main:xi-Q-expansion} follows by a Taylor expansion of
$\log(1 + x)$ around zero, since the ratio differs from $\psifn(\gamma)$
by a vanishing quantity of order $A$). The constant
\begin{equation}
\label{eq:app:main:c-gamma-rho}
  c(\gamma, \rho)
  \;=\;
  \frac{(1 - \gamma)^{-\rho}}{1 - \gamma - \rho}
  \;-\;
  \frac{\psifn(\gamma)^{\rho}}{1 - \gamma - \rho},
\end{equation}
recorded in \textcite[Lemma~1]{DaouiaGirardStupfler2018}, depends on
$(\gamma, \rho)$ only; its explicit form is not used below, only its
finiteness for $\gamma \in (0, 1/2)$ and $\rho \le 0$.

To convert \eqref{eq:app:main:xi-Q-expansion} into a $\sqrt{k}$-rescaled
shift, we show that
\begin{equation}
\label{eq:app:main:Lambda-bullet}
  \sqrt{k}\,A\!\big((1 - \tau_{n})^{-1}\big)
  \;\longrightarrow\;
  \Lambda^{\bullet}
  \;\in\; \R,
\end{equation}
a finite limit determined by the chapter setup. Indeed, for any fixed
$j$, the log-rate convergence $L_{j}(\tau_{n}) \to L_{j}^{\star}$
gives the ratio
\[
  \frac{(1 - \tau_{n})^{-1}}{n_{j}/k_{j}}
  \;=\;
  \frac{k_{j}}{n_{j}(1 - \tau_{n})}
  \;=\;
  e^{L_{j}(\tau_{n})}
  \;\longrightarrow\;
  e^{L_{j}^{\star}}
  \;\in\; (0, \infty),
\]
so by the regular variation $|A| \in \mathrm{RV}_{\rho}$ at infinity
\parencite[Theorem~2.3.9]{deHaanFerreira2006},
\[
  \frac{A((1 - \tau_{n})^{-1})}{A(n_{j}/k_{j})}
  \;\longrightarrow\;
  \big(e^{L_{j}^{\star}}\big)^{\rho}.
\]
Multiplying by $\sqrt{k_{j}}\,A(n_{j}/k_{j}) \to \lambda_{j}$ and then
by $\sqrt{k/k_{j}} \to \sqrt{c_{j}\sum_{i} c_{i}^{-1}}$, the limit
\eqref{eq:app:main:Lambda-bullet} exists and equals
\begin{equation}
\label{eq:app:main:Lambda-bullet-formula}
  \Lambda^{\bullet}
  \;=\;
  \sqrt{c_{j}\,\textstyle\sum_{i = 1}^{m} c_{i}^{-1}}\;
  \lambda_{j}\,\big(e^{L_{j}^{\star}}\big)^{\rho}
  \qquad \text{for every } j = 1, \ldots, m.
\end{equation}
The right-hand side of \eqref{eq:app:main:Lambda-bullet-formula}
appears to depend on $j$, but the left-hand side
$\sqrt{k}\,A((1-\tau_{n})^{-1})$ does not: hence the per-$j$ limits
must coincide whenever they exist. This is a consistency identity
implicit in the chapter setup --- not an additional hypothesis ---
expressing the compatibility of the per-sample bias rates
$\sqrt{k_{j}}\,A(n_{j}/k_{j}) \to \lambda_{j}$ and log-rate limits
$L_{j}(\tau_{n}) \to L_{j}^{\star}$ across $j$ under a single global
$A \in \mathrm{RV}_{\rho}$. Inserting
\eqref{eq:app:main:Lambda-bullet} into
\eqref{eq:app:main:xi-Q-expansion} at $\tau = \tau_{n}$ and using the
sign convention of $(\mathrm{G})$ chosen in
\eqref{eq:app:main:log-decomp},
\begin{equation}
\label{eq:app:main:G-limit}
  \sqrt{k}\,(\mathrm{G})
  \;=\;
  \sqrt{k}\,\big[\log\psifn(\gamma) + \log\Quant_{X}(\tau_{n})
                 - \log\expectile{\tau_{n}}\big]
  \;\longrightarrow\;
  -c(\gamma, \rho)\,\Lambda^{\bullet}
  \;=\; b^{\xi/\Quant},
\end{equation}
which is the finite deterministic shift announced in
\eqref{eq:pool-int:b-xi-Q-def}.

\paragraph{Step (iv): the delta method and the limit.}
Insert \eqref{eq:app:main:H-linearised} and the trivial identity
$\sqrt{k}\,(\mathrm{W}) = \sqrt{k}\,(\mathrm{W})$ into the
$\sqrt{k}$-rescaled version of \eqref{eq:app:main:log-decomp}, and
recall \eqref{eq:app:main:G-limit}:
\begin{equation}
\label{eq:app:main:assembled}
  \sqrt{k}\,\big[\log\hatexpectile{\tau_{n}}^{\mathrm{pool}}
              - \log\expectile{\tau_{n}}\big]
  \;=\;
  \big(m(\gamma),\, 1\big)\,
  \sqrt{k}\!
    \begin{pmatrix}
      \hat\gamma_{n}(\bm\omega_{\gamma}) - \gamma \\[2pt]
      \log\hatWei_{n}(1 - \tau_{n} \mid \bm\omega_{q})
        - \log\Quant_{X}(\tau_{n})
    \end{pmatrix}
  \;+\; \sqrt{k}\,(\mathrm{G})
  \;+\; o_{p}(1).
\end{equation}
By \Cref{prop:pool-int:joint-clt} and the continuous-mapping theorem
applied to the deterministic linear functional $(g, x) \mapsto
m(\gamma)\,g + x$, the leading term on the right of
\eqref{eq:app:main:assembled} converges in distribution to a
univariate Gaussian with mean
$(m(\gamma), 1)\,\bm\mu(\bm\omega_{\gamma}, \bm\omega_{q})$ and
variance
$(m(\gamma), 1)\,\bm\Sigma(\bm\omega_{\gamma}, \bm\omega_{q})\,
 (m(\gamma), 1)^{\top}$. Reading off from \eqref{eq:pool-int:mu-pool}
and \eqref{eq:pool-int:Sigma},
\[
  (m(\gamma), 1)\,\bm\mu(\bm\omega_{\gamma}, \bm\omega_{q})
  \;=\;
  m(\gamma)\,\bm\omega_{\gamma}^{\top}\Bc
  \;+\;
  \bm\omega_{q}^{\top}\Bc^{\star},
\]
and the quadratic form expands to
\begin{align*}
  (m(\gamma), 1)\,\bm\Sigma(\bm\omega_{\gamma}, \bm\omega_{q})\,
                   (m(\gamma), 1)^{\top}
  &\;=\;
  m(\gamma)^{2}\,\bm\omega_{\gamma}^{\top}\Vc\,\bm\omega_{\gamma}
  \;+\;
  2\,m(\gamma)\,
    \bm\omega_{\gamma}^{\top}\,\mathrm{diag}(\bm L^{\star})\,\Vc\,
    \bm\omega_{q} \\
  &\quad{}+\;
  \bm\omega_{q}^{\top}\!\big(\Vc + \mathrm{diag}(\bm L^{\star})^{2}\,
                              \Vc\big)\bm\omega_{q}
  \;=\; \sigma^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q})^{2}.
\end{align*}
The deterministic shift $\sqrt{k}\,(\mathrm{G})$ in
\eqref{eq:app:main:assembled} converges to $b^{\xi/\Quant}$ by
\eqref{eq:app:main:G-limit}. Combining via Slutsky's theorem,
\begin{equation}
\label{eq:app:main:log-CLT}
  \sqrt{k}\,\big[\log\hatexpectile{\tau_{n}}^{\mathrm{pool}}
              - \log\expectile{\tau_{n}}\big]
  \;\todist\;
  \mathcal N\!\big(\mu^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q}),\,
                    \sigma^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q})^{2}
              \big),
\end{equation}
with $\mu^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q}) = m(\gamma)\,
\bm\omega_{\gamma}^{\top}\Bc + \bm\omega_{q}^{\top}\Bc^{\star}
+ b^{\xi/\Quant}$ as in \eqref{eq:pool-int:mu-xi}: the random part
contributes $m(\gamma)\,\bm\omega_{\gamma}^{\top}\Bc +
\bm\omega_{q}^{\top}\Bc^{\star}$ to the mean and the
deterministic shift contributes $+b^{\xi/\Quant}$, both with the same
sign as in \eqref{eq:app:main:assembled}.

To convert the log-ratio statement \eqref{eq:app:main:log-CLT} into
the ratio statement \eqref{eq:pool-int:main-clt}, observe that
\eqref{eq:app:main:log-CLT} forces
$\sqrt{k}\,\log(\hatexpectile{\tau_{n}}^{\mathrm{pool}}/
\expectile{\tau_{n}}) = O_{p}(1)$, equivalently
$\log(\hatexpectile{\tau_{n}}^{\mathrm{pool}}/\expectile{\tau_{n}}) =
o_{p}(1)$. A first-order Taylor expansion of $u \mapsto e^{u} - 1$
around $u = 0$ then gives the exact identity
\begin{equation}
\label{eq:app:main:exp-taylor}
  \frac{\hatexpectile{\tau_{n}}^{\mathrm{pool}}}
       {\expectile{\tau_{n}}} - 1
  \;=\;
  \log\!\frac{\hatexpectile{\tau_{n}}^{\mathrm{pool}}}
              {\expectile{\tau_{n}}}
  \cdot\big(1 + o_{p}(1)\big),
\end{equation}
since the next term in the expansion is squared and thus
$O_{p}(1/k) = o_{p}(1/\sqrt{k})$. Multiplying through by $\sqrt{k}$
and applying Slutsky's theorem,
\[
  \sqrt{k}\!\left(\frac{\hatexpectile{\tau_{n}}^{\mathrm{pool}}}
                          {\expectile{\tau_{n}}} - 1\right)
  \;=\;
  \sqrt{k}\,
    \log\!\frac{\hatexpectile{\tau_{n}}^{\mathrm{pool}}}
                {\expectile{\tau_{n}}}
  \;+\; o_{p}(1)
  \;\todist\;
  \mathcal N\!\big(\mu^{\xi},\, \sigma^{\xi,\,2}\big),
\]
which is \eqref{eq:pool-int:main-clt}. \qed

\section{Proof of \texorpdfstring{\Cref{prop:pool-int:var-opt-distrib,prop:pool-int:var-opt-general}}{Propositions 3.4 and 3.5}}
\label{app:proofs:var-opt-general}

We prove \Cref{prop:pool-int:var-opt-general} (the general case). The
equal-fraction \Cref{prop:pool-int:var-opt-distrib} is recovered as
the specialisation $\bm L^{\star} = L^{\star}\bm 1$ at the end of the
section.



We minimise the asymptotic variance
$\sigma^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q})^{2}$ of
\eqref{eq:pool-int:sigma-xi} over deterministic weight vectors
$\bm\omega_{\gamma}, \bm\omega_{q} \in \R^{m}$ satisfying
$\bm\omega_{\gamma}^{\top}\bm 1 = \bm\omega_{q}^{\top}\bm 1 = 1$,
under the standing assumptions of \Cref{sec:pool-int:setup}. Recall
$\bm D \coloneqq \mathrm{diag}(\bm L^{\star})$ and the scalars
$a, b, c, \Delta$ defined in \eqref{eq:pool-int:var-opt-abcDelta}.

\paragraph{Step (i): the change of variables.}
The factorisation \eqref{eq:pool-int:variance-factorisation},
\[
  \sigma^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q})^{2}
  \;=\;
  \bm u^{\top}\Vc\bm u + \bm\omega_{q}^{\top}\Vc\bm\omega_{q},
  \qquad
  \bm u \coloneqq m(\gamma)\,\bm\omega_{\gamma} + \bm D\,\bm\omega_{q},
\]
relies only on the diagonality of $\Vc$ and is verified by direct
expansion. For $m(\gamma) \ne 0$, the linear map
$(\bm\omega_{\gamma}, \bm\omega_{q}) \mapsto (\bm u, \bm\omega_{q})$
is a bijection of $\R^{m} \times \R^{m}$ onto itself, with inverse
$\bm\omega_{\gamma} = m(\gamma)^{-1}(\bm u - \bm D\,\bm\omega_{q})$.
The simplex constraints translate accordingly: from
$\bm\omega_{\gamma}^{\top}\bm 1 = 1$ and $\bm D\bm 1 = \bm L^{\star}$,
\begin{equation}
\label{eq:app:varopt:constraint-u}
  \bm u^{\top}\bm 1 - \bm\omega_{q}^{\top}\bm L^{\star}
  \;=\; m(\gamma),
  \qquad
  \bm\omega_{q}^{\top}\bm 1 \;=\; 1.
\end{equation}
The optimisation \eqref{eq:pool-int:variance-factorisation} reduces
to a strictly convex quadratic programme on $\R^{2m}$, with the
block-diagonal positive-definite Hessian
$\mathrm{diag}(\Vc, \Vc)$ and the two linear constraints
\eqref{eq:app:varopt:constraint-u}.

\paragraph{Step (ii): the dual variables.}
Introduce Lagrange multipliers $\mu, \nu \in \R$ and form
\[
  \mathcal L(\bm u, \bm\omega_{q}; \mu, \nu)
  \;=\;
  \bm u^{\top}\Vc\bm u + \bm\omega_{q}^{\top}\Vc\bm\omega_{q}
  - \mu\,(\bm u^{\top}\bm 1 - \bm\omega_{q}^{\top}\bm L^{\star}
           - m(\gamma))
  - \nu\,(\bm\omega_{q}^{\top}\bm 1 - 1).
\]
The first-order conditions read
\begin{equation}
\label{eq:app:varopt:foc}
  2\Vc\,\bm u \;=\; \mu\,\bm 1,
  \qquad
  2\Vc\,\bm\omega_{q} \;=\; \nu\,\bm 1 - \mu\,\bm L^{\star}.
\end{equation}
Both equations admit the unique solutions
\begin{equation}
\label{eq:app:varopt:primal-from-dual}
  \bm u \;=\; (\mu/2)\,\Vc^{-1}\bm 1,
  \qquad
  \bm\omega_{q}
  \;=\; (\nu/2)\,\Vc^{-1}\bm 1 - (\mu/2)\,\Vc^{-1}\bm L^{\star},
\end{equation}
displaying $\bm u$ in $\mathrm{span}\{\Vc^{-1}\bm 1\}$ and
$\bm\omega_{q}$ in $\mathrm{span}\{\Vc^{-1}\bm 1, \Vc^{-1}\bm
L^{\star}\}$ at the optimum.

\paragraph{Step (iii): solving for the multipliers.}
Substitute \eqref{eq:app:varopt:primal-from-dual} into the constraints
\eqref{eq:app:varopt:constraint-u}. Recall $a = \bm 1^{\top}\Vc^{-1}
\bm 1$, $b = \bm 1^{\top}\Vc^{-1}\bm L^{\star} = (\bm L^{\star})^{\top}
\Vc^{-1}\bm 1$ (using symmetry of $\Vc^{-1}$), and $c =
(\bm L^{\star})^{\top}\Vc^{-1}\bm L^{\star}$:
\[
  \bm u^{\top}\bm 1 = (\mu/2)\,a,
  \qquad
  \bm\omega_{q}^{\top}\bm L^{\star}
    = (\nu/2)\,b - (\mu/2)\,c,
  \qquad
  \bm\omega_{q}^{\top}\bm 1
    = (\nu/2)\,a - (\mu/2)\,b.
\]
The constraints \eqref{eq:app:varopt:constraint-u} become the linear
system
\begin{equation}
\label{eq:app:varopt:dual-system}
  \begin{pmatrix} a + c & -\,b \\[2pt] -\,b & a \end{pmatrix}\!
  \begin{pmatrix} \mu/2 \\ \nu/2 \end{pmatrix}
  \;=\;
  \begin{pmatrix} m(\gamma) \\ 1 \end{pmatrix}.
\end{equation}
The coefficient matrix has determinant $a(a+c) - b^{2} = \Delta$.
Cauchy--Schwarz in the $\Vc^{-1}$-inner product gives $b^{2}
\le ac$, with equality iff $\bm L^{\star} \in \mathrm{span}\{\bm 1\}$
(which, since all components of $\Vc^{-1}\bm 1$ are strictly positive,
forces $\bm L^{\star} = L^{\star}\bm 1$). Therefore $ac - b^{2} \ge
0$ and, since $a > 0$ (diagonal positive entries of $\Vc^{-1}$),
\begin{equation}
\label{eq:app:varopt:Delta-positive}
  \Delta \;=\; a^{2} + (ac - b^{2}) \;>\; 0,
\end{equation}
strictly, with $\Delta = a^{2}$ exactly under the distributed-inference
setting $\bm L^{\star} = L^{\star}\bm 1$. Cramer's rule applied to
\eqref{eq:app:varopt:dual-system} yields
\begin{equation}
\label{eq:app:varopt:mu-nu}
  \mu/2 \;=\; \frac{m(\gamma)\,a + b}{\Delta},
  \qquad
  \nu/2 \;=\; \frac{a + c + m(\gamma)\,b}{\Delta}.
\end{equation}

\paragraph{Step (iv): the primal solution.}
Inserting \eqref{eq:app:varopt:mu-nu} into
\eqref{eq:app:varopt:primal-from-dual} gives the optimal $\bm
\omega_{q}$ of \eqref{eq:pool-int:var-opt-general-q}. For
$\bm\omega_{\gamma}$, invert the change of variables: $\bm
\omega_{\gamma}^{\star} = m(\gamma)^{-1}(\bm u^{\star} - \bm D\,
\bm\omega_{q}^{\star})$. Using the diagonality identities $\bm D\Vc^{-1}
\bm 1 = \Vc^{-1}\bm L^{\star}$ and $\bm D\Vc^{-1}\bm L^{\star} = \Vc^{-1}
\bm D^{2}\bm 1$ (both consequences of $\Vc, \bm D$ being diagonal and
hence commuting), one obtains
\[
  \bm u^{\star} - \bm D\,\bm\omega_{q}^{\star}
  \;=\;
  \frac{m(\gamma)\,a + b}{\Delta}\,\Vc^{-1}\bm 1
  \;-\;
  \frac{a + c + m(\gamma)\,b}{\Delta}\,\Vc^{-1}\bm L^{\star}
  \;+\;
  \frac{m(\gamma)\,a + b}{\Delta}\,\Vc^{-1}\bm D^{2}\bm 1,
\]
which is \eqref{eq:pool-int:var-opt-general-g} after the factor
$m(\gamma)^{-1}$. The minimum variance follows from Lagrangian
duality: at the optimum of a strictly convex quadratic minimised
under linear constraints, the optimal value equals
$\tfrac{1}{2}(\mu^{\star}\, d_{1} + \nu^{\star}\, d_{2})$ where
$d_{1} = m(\gamma)$, $d_{2} = 1$ are the constraint right-hand sides.
Substituting \eqref{eq:app:varopt:mu-nu}:
\[
  \sigma^{\xi}\!\big(\bm\omega^{\xi,\mathrm{Var}}_{\gamma},
                     \bm\omega^{\xi,\mathrm{Var}}_{q}\big)^{\!2}
  \;=\;
  \frac{m(\gamma)\,[m(\gamma)\,a + b] + [a + c + m(\gamma)\,b]}
       {\Delta}
  \;=\;
  \frac{(m(\gamma)^{2} + 1)\,a + 2\,m(\gamma)\,b + c}{\Delta},
\]
which is \eqref{eq:pool-int:var-opt-general-value}.

\paragraph{Step (v): the departure formula.}
A direct computation gives
\[
  \bm\omega^{\xi,\mathrm{Var}}_{q} - \bm\omega^{\mathrm{Var}}
  \;=\;
  \frac{(a + c + m(\gamma)\,b)\,\Vc^{-1}\bm 1
        - (m(\gamma)\,a + b)\,\Vc^{-1}\bm L^{\star}}{\Delta}
  \;-\;
  \frac{\Vc^{-1}\bm 1}{a}.
\]
Bring to a common denominator $a\Delta$ and collect the
$\Vc^{-1}\bm 1$ coefficient: $a(a + c + m(\gamma)\,b) - \Delta
= a^{2} + ac + m(\gamma)\,ab - a^{2} - ac + b^{2}
= b\,(m(\gamma)\,a + b)$. Hence
\[
  \bm\omega^{\xi,\mathrm{Var}}_{q} - \bm\omega^{\mathrm{Var}}
  \;=\;
  \frac{m(\gamma)\,a + b}{a\,\Delta}\,
  \big[b\,\Vc^{-1}\bm 1 - a\,\Vc^{-1}\bm L^{\star}\big]
  \;=\;
  -\,\frac{m(\gamma)\,a + b}{\Delta}\,
   \Vc^{-1}\big(\bm L^{\star} - \bar L^{\star}\bm 1\big),
\]
with $\bar L^{\star} = b/a$, which is
\eqref{eq:pool-int:var-opt-general-departure}.

\paragraph{Specialisation to the distributed-inference setting.}
Substituting $\bm L^{\star} = L^{\star}\bm 1$ gives $b = L^{\star}a$,
$c = (L^{\star})^{2}a$, $\Delta = a^{2}$. From
\eqref{eq:app:varopt:mu-nu}:
\[
  \mu/2 \;=\; \frac{(m(\gamma) + L^{\star})\,a}{a^{2}}
        \;=\; \frac{m(\gamma) + L^{\star}}{a},
  \qquad
  \nu/2 \;=\; \frac{(1 + L^{\star 2} + m(\gamma)\,L^{\star})\,a}{a^{2}}
        \;=\; \frac{1 + L^{\star 2} + m(\gamma)\,L^{\star}}{a},
\]
and
$\bm\omega^{\star}_{q} = (\nu/2 - (\mu/2)L^{\star})\,\Vc^{-1}\bm 1
= ((1 + L^{\star 2} + m(\gamma)L^{\star}) - L^{\star}(m(\gamma)
+ L^{\star}))\Vc^{-1}\bm 1/a = \Vc^{-1}\bm 1/a =
\bm\omega^{\mathrm{Var}}$. A symmetric calculation gives
$\bm\omega^{\star}_{\gamma} = \bm\omega^{\mathrm{Var}}$, and the
minimum-variance value reduces to $[(m(\gamma) + L^{\star})^{2} +
1]/a$, recovering
\Cref{prop:pool-int:var-opt-distrib}.

\paragraph{The degenerate case $m(\gamma) = 0$.}
When $m(\gamma) = 0$, the change-of-variables map is no longer a
bijection: $\bm u = \bm D\bm\omega_{q}$ depends only on $\bm\omega_{q}$
and $\bm\omega_{\gamma}$ disappears from the variance. The factorisation
collapses to
$\sigma^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q})^{2}
= \bm\omega_{q}^{\top}\Vc(\bm I + \bm D^{2})\bm\omega_{q}$,
a strictly convex quadratic in $\bm\omega_{q}$ alone (since $\Vc(\bm I
+ \bm D^{2})$ is the product of two positive-definite diagonal matrices,
hence positive definite). Minimisation under $\bm\omega_{q}^{\top}\bm 1
= 1$ gives, by the standard Lagrangian calculation for a single linear
constraint,
\begin{equation}
\label{eq:app:varopt:m-zero}
  \bm\omega^{\xi,\mathrm{Var}}_{q}\big|_{m(\gamma) = 0}
  \;=\;
  \frac{(\bm I + \bm D^{2})^{-1}\Vc^{-1}\bm 1}
       {\bm 1^{\top}(\bm I + \bm D^{2})^{-1}\Vc^{-1}\bm 1},
\end{equation}
with $\bm\omega_{\gamma}$ indeterminate. In the distributed-inference
setting $\bm D = L^{\star}\bm I$, $(\bm I + \bm D^{2})^{-1}$ is a
scalar multiple of $\bm I$ and \eqref{eq:app:varopt:m-zero} reduces to
$\bm\omega^{\mathrm{Var}}$; in general it differs from
$\bm\omega^{\mathrm{Var}}$ along directions in
$\mathrm{span}\{\bm 1, \bm L^{\star},\ldots\}$. \qed

\section{Proof of \texorpdfstring{\Cref{prop:pool-int:amse-opt}}{Proposition 3.6}}
\label{app:proofs:amse-opt}

We prove existence, uniqueness, and the closed form
\eqref{eq:pool-int:amse-opt-formula} of the AMSE minimiser under the
standing assumptions of \Cref{sec:pool-int:setup} and $m(\gamma) \ne 0$.
Recall the stacked notation
$\bm w = (\bm\omega_{\gamma}^{\top}, \bm\omega_{q}^{\top})^{\top}$,
the bias gradient and variance Hessian $\bm h, \bm Q$ of
\eqref{eq:pool-int:amse-stacked}, the constraint matrix $\bm A \in
\R^{2m \times 2}$ with columns $\bm 1_{g} = (\bm 1^{\top}, \bm
0^{\top})^{\top}$ and $\bm 1_{q} = (\bm 0^{\top}, \bm 1^{\top})^{\top}$,
and the matrix $\bm M \coloneqq \bm Q + \bm h\bm h^{\top}$.

\paragraph{Step (i): rewriting the AMSE as a quadratic form.}
By construction $\mu^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q}) =
\bm h^{\top}\bm w + b^{\xi/\Quant}$ and
$\sigma^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q})^{2} = \bm w^{\top}\bm
Q\bm w$, so
\begin{align*}
  \mathrm{AMSE}^{\xi}(\bm w)
  &\;=\;
  (\bm h^{\top}\bm w + b^{\xi/\Quant})^{2}
  + \bm w^{\top}\bm Q\bm w
  \\
  &\;=\;
  \bm w^{\top}\bm M\,\bm w
  \;+\; 2\,b^{\xi/\Quant}\,\bm h^{\top}\bm w
  \;+\; (b^{\xi/\Quant})^{2},
\end{align*}
recognising $(\bm h^{\top}\bm w)^{2} = \bm w^{\top}(\bm h\bm h^{\top})
\bm w$ and $\bm M = \bm Q + \bm h\bm h^{\top}$. The additive constant
$(b^{\xi/\Quant})^{2}$ does not influence the argmin and is dropped
below.

\paragraph{Step (ii): positive-definiteness of $\bm M$.}
$\bm Q$ is positive definite. Indeed, the variance factorisation
$\bm w^{\top}\bm Q\bm w = \bm u^{\top}\Vc\bm u + \bm\omega_{q}^{\top}
\Vc\bm\omega_{q}$ with $\bm u = m(\gamma)\bm\omega_{\gamma} + \bm
D\bm\omega_{q}$ exhibits $\bm Q$ as the pull-back of the
block-diagonal positive-definite matrix $\mathrm{diag}(\Vc, \Vc)$
under the linear bijection $\bm w \mapsto (\bm u, \bm\omega_{q})$
(valid for $m(\gamma) \ne 0$, see \Cref{app:proofs:var-opt-general}).
Hence $\bm w^{\top}\bm Q\bm w \ge 0$ with equality iff $\bm u = 0$
and $\bm\omega_{q} = 0$, which forces $\bm w = 0$. The rank-one
positive semi-definite update $\bm h\bm h^{\top}$ preserves positive
definiteness, so $\bm M = \bm Q + \bm h\bm h^{\top}$ is positive
definite. Strict convexity of $\mathrm{AMSE}^{\xi}(\bm w) -
(b^{\xi/\Quant})^{2}$ on $\R^{2m}$, and a fortiori on the closed
affine constraint set $\{\bm A^{\top}\bm w = \bm 1_{2}\}$, follows.

\paragraph{Step (iii): existence and uniqueness via the Lagrangian.}
Form
\[
  \mathcal L(\bm w; \bm\nu)
  \;=\;
  \bm w^{\top}\bm M\,\bm w
  + 2\,b^{\xi/\Quant}\,\bm h^{\top}\bm w
  - \bm\nu^{\top}(\bm A^{\top}\bm w - \bm 1_{2}),
\]
with $\bm\nu \in \R^{2}$. The first-order conditions read
\begin{equation}
\label{eq:app:amse:foc-stacked}
  2\,\bm M\,\bm w + 2\,b^{\xi/\Quant}\,\bm h - \bm A\,\bm\nu
  \;=\; \bm 0,
\end{equation}
together with $\bm A^{\top}\bm w = \bm 1_{2}$. From
\eqref{eq:app:amse:foc-stacked},
\begin{equation}
\label{eq:app:amse:primal-from-dual}
  \bm w
  \;=\;
  \bm M^{-1}\!\left[\tfrac{1}{2}\bm A\,\bm\nu
                    - b^{\xi/\Quant}\,\bm h\right].
\end{equation}
Imposing the constraint $\bm A^{\top}\bm w = \bm 1_{2}$:
\[
  \tfrac{1}{2}\,\bm A^{\top}\bm M^{-1}\bm A\,\bm\nu
  - b^{\xi/\Quant}\,\bm A^{\top}\bm M^{-1}\bm h
  \;=\; \bm 1_{2}.
\]
The matrix $\bm A^{\top}\bm M^{-1}\bm A \in \R^{2 \times 2}$ is
positive definite: $\bm M^{-1}$ is PD, and $\bm A$ has full column
rank since its two columns $\bm 1_{g}, \bm 1_{q}$ have disjoint
nonzero supports. Hence
\begin{equation}
\label{eq:app:amse:nu-formula}
  \bm\nu \;=\; 2\,(\bm A^{\top}\bm M^{-1}\bm A)^{-1}\!
              \big(\bm 1_{2} \;+\; b^{\xi/\Quant}\,
                   \bm A^{\top}\bm M^{-1}\bm h\big).
\end{equation}
Substituting \eqref{eq:app:amse:nu-formula} back into
\eqref{eq:app:amse:primal-from-dual} delivers the closed form
\eqref{eq:pool-int:amse-opt-formula}. Strict convexity of the
objective on the affine constraint set together with the linear FOC
\eqref{eq:app:amse:foc-stacked} forces this $\bm w$ to be the unique
constrained minimiser.

\paragraph{Step (iv): equivalence with the block FOC system.}
Expand \eqref{eq:app:amse:foc-stacked} block-by-block. The product
$\bm M\bm w = \bm Q\bm w + (\bm h^{\top}\bm w)\,\bm h = \bm Q\bm w +
(\mu^{\xi} - b^{\xi/\Quant})\bm h$, using $\mu^{\xi} = \bm h^{\top}
\bm w + b^{\xi/\Quant}$. Substituting,
\eqref{eq:app:amse:foc-stacked} becomes
\[
  2\,\bm Q\bm w \;+\; 2\,(\mu^{\xi} - b^{\xi/\Quant})\,\bm h
  \;+\; 2\,b^{\xi/\Quant}\,\bm h
  \;=\; \bm A\,\bm\nu
  \quad\Longleftrightarrow\quad
  \bm Q\bm w \;+\; \mu^{\xi}\,\bm h \;=\; \tfrac{1}{2}\,\bm A\,\bm\nu.
\]
Writing $\bm\nu = (\lambda_{\gamma}, \lambda_{q})^{\top}$ and reading
off the upper and lower blocks of size $m$:
\begin{align*}
  \big[\bm Q\bm w\big]_{1, \ldots, m}
  &\;=\;
  m(\gamma)^{2}\,\Vc\,\bm\omega_{\gamma}
  + m(\gamma)\,\bm D\,\Vc\,\bm\omega_{q},
  \\
  \big[\bm Q\bm w\big]_{m+1, \ldots, 2m}
  &\;=\;
  m(\gamma)\,\bm D\,\Vc\,\bm\omega_{\gamma}
  + (\Vc + \bm D^{2}\Vc)\,\bm\omega_{q},
\end{align*}
$[\bm h]_{1,\ldots,m} = m(\gamma)\Bc$,
$[\bm h]_{m+1,\ldots,2m} = \Bc^{\star}$,
$[\bm A\bm\nu]_{1,\ldots,m} = \lambda_{\gamma}\bm 1$,
$[\bm A\bm\nu]_{m+1,\ldots,2m} = \lambda_{q}\bm 1$.
The block decomposition reproduces \eqref{eq:pool-int:amse-foc}.
Uniqueness from strict convexity, combined with the consistency
identity $\mu^{\xi} = m(\gamma)\,\bm\omega_{\gamma}^{\top}\Bc +
\bm\omega_{q}^{\top}\Bc^{\star} + b^{\xi/\Quant}$ at the minimiser,
completes the equivalence. \qed

\section{Proof of \texorpdfstring{\Cref{cor:pool-int:CI}}{Corollary 3.7}}
\label{app:proofs:CI}

We prove the asymptotic-coverage statement of
\Cref{cor:pool-int:CI} under the standing assumptions of
\Cref{sec:pool-int:setup}, the undersmoothing condition
\eqref{eq:pool-int:undersmoothing}, and the consistency assumptions
\eqref{eq:pool-int:CI-consistency} on the estimated weight pair
$(\hat{\bm\omega}_{\gamma}, \hat{\bm\omega}_{q})$, the plug-in
covariance $\hatVc$, and the plug-in QB Jacobian $\hat m$. We retain
the simplex constraints
$\hat{\bm\omega}_{\gamma}^{\top}\bm 1 =
\hat{\bm\omega}_{q}^{\top}\bm 1 = 1$ \emph{exactly}, which holds for
the variance- and AMSE-optimal plug-ins of
\Cref{prop:bg:optimal-weights,prop:pool-int:var-opt-general,prop:pool-int:amse-opt}
by construction.

The proof composes four steps:
(i) recast the conclusion of \Cref{thm:pool-int:main} on the log
scale;
(ii) lift the deterministic-weight CLT to the estimated weights via
the exact simplex constraint and Slutsky;
(iii) substitute the plug-in variance to obtain the asymptotic pivot
under undersmoothing;
(iv) transport coverage to the original scale by strict monotonicity
of $\exp$.

\paragraph{Step (i): log-scale recasting.}
By the QB factorisation \eqref{eq:pool-int:expectile-pool},
\begin{equation}
\label{eq:app:CI:log-decomposition}
  \log\hatexpectile{\tau_{n}}^{\mathrm{pool}}\!
       (\bm\omega_{\gamma}, \bm\omega_{q})
  \;=\;
  \log\psifn\!\big(\hat\gamma_{n}(\bm\omega_{\gamma})\big)
  \;+\;
  \log\hatWei_{n}(1 - \tau_{n} \mid \bm\omega_{q}).
\end{equation}
\Cref{thm:pool-int:main} delivers $\sqrt{k}\,
(\hatexpectile{\tau_{n}}^{\mathrm{pool}}(\bm\omega_{\gamma},
\bm\omega_{q})/\expectile{\tau_{n}} - 1) \todist
\mathcal N(\mu^{\xi}, \sigma^{\xi,\,2})$. The displayed log-CLT
\eqref{eq:app:main:log-CLT} in the proof of
\Cref{thm:pool-int:main} (\Cref{app:proofs:main-clt}) is the same
statement on the log scale,
\begin{equation}
\label{eq:app:CI:log-clt-det}
  \sqrt{k}\,
  \big(\log\hatexpectile{\tau_{n}}^{\mathrm{pool}}\!
            (\bm\omega_{\gamma}, \bm\omega_{q})
       - \log\expectile{\tau_{n}}\big)
  \;\todist\;
  \mathcal N\!\big(\mu^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q}),\,
                    \sigma^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q})^{2}\big).
\end{equation}
Under undersmoothing \eqref{eq:pool-int:undersmoothing}, every entry
of $\Bc, \Bc^{\star}$ and the offset $b^{\xi/\Quant}$ vanishes
(inspection of \eqref{eq:pool-int:Bcstar-def},
\eqref{eq:pool-int:b-xi-Q-def}, and the formula for $\Bc$ in
\Cref{thm:bg:pool-hill}, each of which is a continuous linear
combination of the per-sample bias rates
$\lambda_{j} = \lim \sqrt{k_{j}}\,A(n_{j}/k_{j})$). Hence
$\mu^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q}) = 0$ and the limit in
\eqref{eq:app:CI:log-clt-det} reduces to
$\mathcal N(0, \sigma^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q})^{2})$.

\paragraph{Step (ii): estimated-weight lift.}
We show that replacing $(\bm\omega_{\gamma}, \bm\omega_{q})$ by
$(\hat{\bm\omega}_{\gamma}, \hat{\bm\omega}_{q})$ in
\eqref{eq:app:CI:log-decomposition} changes the left-hand side of
\eqref{eq:app:CI:log-clt-det} by $o_{p}(1)$. Decompose this change
into a Hill and a Weissman contribution,
\begin{align*}
  \Delta_{n}^{\gamma}
  &\;\coloneqq\;
    \log\psifn\!\big(\hat\gamma_{n}(\hat{\bm\omega}_{\gamma})\big)
    - \log\psifn\!\big(\hat\gamma_{n}(\bm\omega_{\gamma})\big),
  \\[2pt]
  \Delta_{n}^{q}
  &\;\coloneqq\;
    \log\hatWei_{n}(1 - \tau_{n} \mid \hat{\bm\omega}_{q})
    - \log\hatWei_{n}(1 - \tau_{n} \mid \bm\omega_{q}).
\end{align*}

\emph{Hill contribution.}
Linearity of the pooled Hill in its weights
(\eqref{eq:pool-int:hill-pool}) gives
\[
  \hat\gamma_{n}(\hat{\bm\omega}_{\gamma})
    - \hat\gamma_{n}(\bm\omega_{\gamma})
  \;=\;
  (\hat{\bm\omega}_{\gamma} - \bm\omega_{\gamma})^{\top}
    \hat{\bm\gamma}_{n}
  \;=\;
  (\hat{\bm\omega}_{\gamma} - \bm\omega_{\gamma})^{\top}
    \big(\hat{\bm\gamma}_{n} - \gamma\bm 1\big),
\]
the deterministic offset $\gamma\bm 1$ vanishing by the exact simplex
identity
$(\hat{\bm\omega}_{\gamma} - \bm\omega_{\gamma})^{\top}\bm 1
= 1 - 1 = 0$ on both the plug-in and the limit weight vectors. The
per-sample Hill CLT (within-sample component of
\eqref{eq:pool-int:per-sample-clt}) gives
$\sqrt{k_{j}}(\hat\gamma_{j} - \gamma) = O_{p}(1)$, hence
$\sqrt{k}(\hat\gamma_{j} - \gamma) = \sqrt{k/k_{j}}\,O_{p}(1) =
O_{p}(1)$ via $k_{j}/k \to c_{j} \in (0, 1)$ from
\eqref{eq:bg:pool-rates}. Therefore $\sqrt{k}(\hat{\bm\gamma}_{n}
- \gamma\bm 1) = O_{p}(1)$, and the Cauchy--Schwarz inequality
combined with $\hat{\bm\omega}_{\gamma} - \bm\omega_{\gamma} =
o_{p}(1)$ yields
\begin{equation}
\label{eq:app:CI:hill-lift}
  \sqrt{k}\,\big(\hat\gamma_{n}(\hat{\bm\omega}_{\gamma})
                  - \hat\gamma_{n}(\bm\omega_{\gamma})\big)
  \;=\;
  (\hat{\bm\omega}_{\gamma} - \bm\omega_{\gamma})^{\top}\,
  \sqrt{k}\big(\hat{\bm\gamma}_{n} - \gamma\bm 1\big)
  \;=\; o_{p}(1) \cdot O_{p}(1) \;=\; o_{p}(1).
\end{equation}
The map $\log\psifn$ is continuously differentiable on $(0, 1/2)$
with derivative $m(\,\cdot\,)$ defined in
\eqref{eq:pool-int:m-function}; on the good event
$\{\hat\gamma_{n}(\bm\omega_{\gamma}),
\hat\gamma_{n}(\hat{\bm\omega}_{\gamma}) \in (0, 1)\}$ --- of
probability $\to 1$ by the consistency
$\hat\gamma_{n}(\hat{\bm\omega}_{\gamma}),
\hat\gamma_{n}(\bm\omega_{\gamma}) \toprob \gamma$ --- the mean
value theorem yields a random $\tilde\gamma_{n}$ between the two
pooled values such that
\[
  \Delta_{n}^{\gamma}
  \;=\;
  m(\tilde\gamma_{n})\,
  \big[\hat\gamma_{n}(\hat{\bm\omega}_{\gamma})
       - \hat\gamma_{n}(\bm\omega_{\gamma})\big].
\]
Squeeze gives $\tilde\gamma_{n} \toprob \gamma$ and, by continuity of
$m(\,\cdot\,)$, $m(\tilde\gamma_{n}) \toprob m(\gamma)$. Combining
with \eqref{eq:app:CI:hill-lift}, $\sqrt{k}\,\Delta_{n}^{\gamma}
= O_{p}(1) \cdot o_{p}(1) = o_{p}(1)$.

\emph{Weissman contribution.}
Weighted-geometric linearity of $\log\hatWei_{n}$ in $\bm\omega_{q}$
(\eqref{eq:pool-int:weissman-pool}) gives
\[
  \Delta_{n}^{q}
  \;=\;
  (\hat{\bm\omega}_{q} - \bm\omega_{q})^{\top}
    \big(\log\hat{\bm q}^{\star}_{n}
         - \log\Quant_{X}(\tau_{n})\bm 1\big),
\]
where the deterministic offset $\log\Quant_{X}(\tau_{n})\bm 1$
cancels via the simplex identity
$(\hat{\bm\omega}_{q} - \bm\omega_{q})^{\top}\bm 1 = 0$. The
within-sample Weissman component of
\eqref{eq:pool-int:per-sample-clt} delivers
$\sqrt{k_{j}}(\log\hat q^{\star}_{j} - \log\Quant_{X}(\tau_{n}))
= O_{p}(1)$ under $L_{j}(\tau_{n}) \to L_{j}^{\star} \in \R$, and the
same $\sqrt{k/k_{j}} = O(1)$ rescaling promotes this to
$\sqrt{k}(\log\hat{\bm q}^{\star}_{n}
- \log\Quant_{X}(\tau_{n})\bm 1) = O_{p}(1)$. Hence
\begin{equation}
\label{eq:app:CI:weissman-lift}
  \sqrt{k}\,\Delta_{n}^{q}
  \;=\;
  (\hat{\bm\omega}_{q} - \bm\omega_{q})^{\top}\,
  \sqrt{k}\big(\log\hat{\bm q}^{\star}_{n}
               - \log\Quant_{X}(\tau_{n})\bm 1\big)
  \;=\; o_{p}(1) \cdot O_{p}(1) \;=\; o_{p}(1).
\end{equation}

\emph{Assembly.}
Combining \eqref{eq:app:CI:log-clt-det} (under undersmoothing,
limit-mean zero) with $\sqrt{k}\,\Delta_{n}^{\gamma}
+ \sqrt{k}\,\Delta_{n}^{q} = o_{p}(1)$ and Slutsky's theorem,
\begin{equation}
\label{eq:app:CI:log-clt-plugin}
  \sqrt{k}\,
  \big(\log\hatexpectile{\tau_{n}}^{\mathrm{pool}}\!
            (\hat{\bm\omega}_{\gamma}, \hat{\bm\omega}_{q})
       - \log\expectile{\tau_{n}}\big)
  \;\todist\;
  \mathcal N\!\big(0,\,
                    \sigma^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q})^{2}\big).
\end{equation}

\paragraph{Step (iii): self-normalisation.}
The variance \eqref{eq:pool-int:sigma-xi} is a polynomial in
$(\Vc, \bm L^{\star}, m, \bm\omega_{\gamma}, \bm\omega_{q})$, and
$\sigma^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q})^{2} > 0$ because
$\Vc$ is positive definite, $\bm\omega_{q}^{\top}\bm 1 = 1$ forces
$\bm\omega_{q} \ne \bm 0$, and the structurally non-negative
contribution $\bm\omega_{q}^{\top}\Vc\bm\omega_{q}$ to
\eqref{eq:pool-int:sigma-xi} is strictly positive. The plug-in
substitutes $\hatVc, \hat L_{j}^{\star} \coloneqq L_{j}(\tau_{n}),
\hat m, \hat{\bm\omega}_{\gamma}, \hat{\bm\omega}_{q}$ for their
population analogues. The log-rate plug-in
$\hat L_{j}^{\star} = L_{j}(\tau_{n}) =
\log(k_{j}/(n_{j}(1 - \tau_{n})))$ is deterministic and converges to
$L_{j}^{\star}$ by the chapter setup of \Cref{sec:pool-int:setup}; the
remaining four convergences are the assumptions
\eqref{eq:pool-int:CI-consistency}. The continuous-mapping theorem
applied first to the polynomial \eqref{eq:pool-int:sigma-xi} and then
to $\sqrt{\,\cdot\,}$ on $(0, \infty)$ yields
\[
  \hat\sigma^{\xi}(\hat{\bm\omega}_{\gamma}, \hat{\bm\omega}_{q})
  \;\toprob\;
  \sigma^{\xi}(\bm\omega_{\gamma}, \bm\omega_{q}) \;>\; 0.
\]
Slutsky's theorem applied to the ratio of
\eqref{eq:app:CI:log-clt-plugin} and this consistent variance
estimator produces the asymptotic pivot
\begin{equation}
\label{eq:app:CI:pivot}
  Z_{n}
  \;\coloneqq\;
  \frac{\sqrt{k}\,
        \big(\log\hatexpectile{\tau_{n}}^{\mathrm{pool}}\!
              (\hat{\bm\omega}_{\gamma}, \hat{\bm\omega}_{q})
             - \log\expectile{\tau_{n}}\big)}
       {\hat\sigma^{\xi}(\hat{\bm\omega}_{\gamma}, \hat{\bm\omega}_{q})}
  \;\todist\; \mathcal N(0, 1).
\end{equation}
For every $\alpha \in (0, 1)$,
$\Prob(|Z_{n}| \le z_{1 - \alpha/2}) \to 1 - \alpha$. Inverting the
inequality $|Z_{n}| \le z_{1 - \alpha/2}$ for
$\log\expectile{\tau_{n}}$ recovers exactly the event
$\log\expectile{\tau_{n}} \in I_{n}^{\log}$ with $I_{n}^{\log}$
defined in \eqref{eq:pool-int:CI-log}, so
$\Prob(\log\expectile{\tau_{n}} \in I_{n}^{\log}) \to 1 - \alpha$.

\paragraph{Step (iv): transport to the original scale.}
The map $x \mapsto e^{x}$ is strictly increasing and continuous on
$\R$, so for the random closed interval
$I_{n}^{\log} = [a_{n}, b_{n}]$,
\[
  \log\expectile{\tau_{n}} \in [a_{n}, b_{n}]
  \quad \Longleftrightarrow \quad
  \expectile{\tau_{n}} \in [e^{a_{n}}, e^{b_{n}}]
  \;=\; \exp(I_{n}^{\log})
  \;=\; I_{n}.
\]
The two events coincide pointwise on the underlying probability space,
hence
$\Prob(\expectile{\tau_{n}} \in I_{n})
= \Prob(\log\expectile{\tau_{n}} \in I_{n}^{\log}) \to 1 - \alpha$. \qed
