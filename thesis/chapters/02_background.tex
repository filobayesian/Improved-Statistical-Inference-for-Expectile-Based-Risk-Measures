\chapter{Background}
\label{ch:background}

\section{Regular variation and the second-order condition}
\label{sec:bg:rv}

Throughout the thesis we work with a real-valued random loss $X$ that
has continuous distribution function $F$ and survival function
$\overline F = 1 - F$. Given an iid sample $X_{1}, \ldots, X_{n}$ from
$F$, the corresponding order statistics are denoted by
$X_{1:n} \le X_{2:n} \le \cdots \le X_{n:n}$.

Heavy-tailed asymptotics are governed by the language of regular
variation; the reference is \textcite[Appendix~B.1]{deHaanFerreira2006}.

\begin{definition}[Regular variation]
\label{def:bg:rv}
A Lebesgue measurable function $f : (0,\infty) \to \R$ that is eventually
positive is \emph{regularly varying at infinity with index} $\alpha \in \R$, written
$f \in \mathrm{RV}_{\alpha}$, if for every $y > 0$,
\[
  \lim_{t \to \infty} \frac{f(t y)}{f(t)} \;=\; y^{\alpha}.
\]
The functions in $\mathrm{RV}_{0}$ are called \emph{slowly varying}.
\end{definition}

The first-order condition under which all of our asymptotics are taken
asserts that $F$ has a Pareto-type tail.

\begin{condition}[First-order condition]
\label{cond:bg:first-order}
There exists $\gamma > 0$ such that
$\overline F \in \mathrm{RV}_{-1/\gamma}$, or equivalently the
\emph{tail quantile function}
\[
  U(t) \;\coloneqq\; \inf\!\Big\{ x \in \R \,:\, \tfrac{1}{\overline F(x)} \ge t \Big\}
  \,,\qquad t>1,
\]
satisfies $U \in \mathrm{RV}_{\gamma}$.
\end{condition}

Throughout, for a continuous distribution function $F$, we write
\[
  \Quant_X(\tau)
  \;\coloneqq\;
  F^{\leftarrow}(\tau)
  =
  \inf\{x\in\R:F(x)\ge \tau\},
  \qquad 0<\tau<1.
\]
With the tail quantile $U(t)=F^{\leftarrow}(1-1/t)$, $t>1$, this gives
\[
  \Quant_X(\tau)=U((1-\tau)^{-1}),
  \qquad
  \Quant_X(1-p)=U(1/p).
\]

\Cref{cond:bg:first-order} is the Fr\'echet max-domain condition with
extreme-value index $\gamma$ \parencite[Theorem~1.2.1]{deHaanFerreira2006};
standard examples include Pareto, Burr, Student-$t$, Fr\'echet and
$F$-distribution tails \parencite[\S2.6]{deHaanFerreira2006}. For
normal limits we use the standard second-order refinement.

\begin{condition}[Second-order condition $\mathcal C_{2}(\gamma, \rho, A)$]
\label{cond:bg:second-order}
There exist $\gamma > 0$, a second-order parameter $\rho \le 0$, and a
measurable auxiliary function $A$ of constant sign with $A(t) \to 0$ as
$t \to \infty$, such that for every $y > 0$,
\[
  \lim_{t \to \infty} \frac{1}{A(t)}
    \!\left[ \frac{U(t y)}{U(t)} - y^{\gamma}\right]
  \;=\; y^{\gamma}\, \frac{y^{\rho} - 1}{\rho},
\]
where the right-hand side is read as $y^{\gamma} \log y$ when
$\rho = 0$.
\end{condition}

The function $|A|$ is itself regularly varying with index $\rho$
\parencite[Theorem~2.3.9]{deHaanFerreira2006}. Since $A$ has constant
sign eventually, signed ratios $A(t y)/A(t)$ converge to $y^\rho$; the
rate $A$ is the source of the second-order bias terms below.

\begin{condition}[Finite first-moment condition]
\label{cond:bg:finite-first-moment}
The extreme-value index satisfies $\gamma \in (0,1)$ and the negative
part $X^{-}=\max(-X,0)$ satisfies
\[
  \E X^{-}<\infty .
\]
\end{condition}

Under \Cref{cond:bg:first-order}, $\gamma<1$ gives $\E X^{+}<\infty$;
together with the displayed lower-tail condition this is the finite-mean
regime for the quantile-based expectile approximation.

\begin{condition}[LAWS lower-tail moment condition]
\label{cond:bg:lower-tail-moment}
For some $\delta>0$, the negative part
$X^{-}=\max(-X,0)$ satisfies
\[
  \E (X^{-})^{2+\delta}<\infty .
\]
\end{condition}

This stronger finite-variance-type condition is used only for direct
LAWS expectile estimation. It is not part of the quantile-based plug-in
route used in the contribution chapters.

\section{Tail-index and extreme-quantile estimation}
\label{sec:bg:tail-index-quantile}

We collect the two classical estimators used throughout the thesis: the
Hill estimator of $\gamma$ and the Weissman extrapolation for extreme
quantiles. The textbook reference is
\textcite[Ch.~3--4]{deHaanFerreira2006}.

\subsection{The Hill estimator}
\label{sec:bg:hill}

Given an iid sample $X_{1}, \ldots, X_{n}$ from $F$ and an integer
$k = k_{n} \in \{1, \ldots, n-1\}$, the \emph{Hill estimator}
\parencite{Hill1975} of
$\gamma$ based on the $k + 1$ top order statistics is
\begin{equation}
\label{eq:bg:hill}
  \hat\gamma_{n}(k) \;=\; \frac{1}{k} \sum_{i=1}^{k} \log\!\left(
    \frac{X_{n-i+1\,:\,n}}{X_{n-k\,:\,n}} \right).
\end{equation}
Since $U(t)\to\infty$, $X_{n-k:n}>0$ with probability tending to one
for intermediate $k$; log-based estimators are understood on this event.
For tail-index estimation one may also shift the data, but shifts must
be undone for quantile or expectile estimation. The consistency result
below is \textcite[Theorem~3.2.2]{deHaanFerreira2006}; the normal limit
is \textcite[Theorem~3.2.5]{deHaanFerreira2006} when $\rho<0$, with the
$\rho=0$ endpoint covered by the $m=1$ specialisation of
\textcite[Theorem~1]{DaouiaPadoanStupfler2021}.

\begin{theorem}
\label{thm:bg:hill}
Assume $\mathcal C_{2}(\gamma, \rho, A)$ with $\gamma > 0$. Let
$k = k_{n} \to \infty$ and $k / n \to 0$ as $n \to \infty$. Then
$\hat\gamma_{n}(k) \toprob \gamma$. If, in addition,
$\sqrt{k}\, A(n/k) \to \lambda \in \R$, then
\[
  \sqrt{k}\,\big( \hat\gamma_{n}(k) - \gamma \big)
  \;\todist\;
  \mathcal N\!\left( \frac{\lambda}{1 - \rho},\; \gamma^{2} \right).
\]
\end{theorem}

The choice of $k$ controls the usual bias--variance trade-off; here it
is treated as fixed by the data analyst.

\subsection{Weissman extrapolation to extreme quantiles}
\label{sec:bg:weissman}

The \emph{Weissman extrapolation} \parencite{Weissman1978} extrapolates
from the intermediate-order statistic $X_{n - k:n}$ --- which
estimates the quantile $\Quant_{X}(1 - k/n)$ at the intermediate level
$1 - k/n$ --- to a quantile at a very-extreme level $1 - p$ with
$p = p_{n} \to 0$ as $n \to \infty$. Plugging the Hill estimator into
the first-order identity $U(t y) / U(t) \to y^{\gamma}$ gives the
Weissman estimator
\begin{equation}
\label{eq:bg:weissman}
  \hatWei_{n}(1 - p \mid k)
  \;=\; \left( \frac{k}{n p} \right)^{\hat\gamma_{n}(k)}
        X_{n - k\,:\,n}.
\end{equation}
The theorem below, the Hill-estimator specialisation of
\textcite[Theorem~4.3.8]{deHaanFerreira2006}, covers the far-tail
regime $np=o(k)$; the case $np=O(1)$ is very-extreme in the sense that
only finitely many observations are expected beyond the target.

\begin{theorem}
\label{thm:bg:weissman}
Assume $\mathcal C_{2}(\gamma, \rho, A)$ with $\gamma > 0$ and
$\rho < 0$. Let $k = k_{n} \to \infty$, $k / n \to 0$, $p = p_{n} \to 0$
with $k / (np) \to \infty$ and $\sqrt{k} / \log(k / (np)) \to \infty$.
Suppose $\sqrt{k}\, A(n/k) \to \lambda \in \R$. Then
\[
  \frac{\sqrt{k}}{\log\!\big(k/(np)\big)}
  \left( \frac{\hatWei_{n}(1 - p \mid k)}{\Quant_{X}(1 - p)} - 1 \right)
  \;\todist\;
  \mathcal N\!\left( \frac{\lambda}{1 - \rho},\; \gamma^{2} \right).
\]
\end{theorem}

The log-denominator is the cost of extrapolation. Up to that factor the
limit is the Hill limit, which is why
\textcite[\S2.3]{DaouiaPadoanStupfler2021} pool Weissman estimators
geometrically. Whether the rebuilt expectile estimator has exactly this
first-order law is the question isolated in \Cref{ch:pool-ext}.

\section{Expectiles}
\label{sec:bg:expectiles}

Expectiles were introduced by \textcite{NeweyPowell1987} as
asymmetric-least-squares analogues of quantiles. Their role as coherent
and elicitable risk measures is developed in
\textcite{BelliniKlarMullerRosazzaGianin2014,Ziegel2016,Gneiting2011,EhmGneitingJordanKrueger2016}.

\subsection{Definition}

Let $X$ be a real-valued random variable with $\E |X| < \infty$, and
fix a level $\tau \in (0, 1)$. The $\tau$-\emph{expectile} of $X$ is
defined as
\begin{equation}
\label{eq:bg:expectile-def}
  \expectile{\tau}(X)
  \;\coloneqq\;
  \argmin_{\theta \in \R}\;
  \E\!\left[\, \eta_{\tau}(X - \theta) - \eta_{\tau}(X) \,\right],
  \qquad
  \eta_{\tau}(u) \coloneqq \abs{\tau - \one\{u \le 0\}}\, u^{2}.
\end{equation}
The subtraction of $\eta_\tau(X)$ makes the criterion finite under
$\E|X|<\infty$ and does not affect the minimiser; at $\tau=1/2$ one
recovers the mean.

The convex objective \eqref{eq:bg:expectile-def} has a unique
minimiser, and its first-order condition reduces to the implicit
equation
\begin{equation}
\label{eq:bg:expectile-foc}
  \tau\, \E\!\big[(X - \expectile{\tau}(X))_{+}\big]
  \;=\;
  (1 - \tau)\, \E\!\big[(\expectile{\tau}(X) - X)_{+}\big],
\end{equation}
where $u_{+} = \max(u, 0)$. Thus expectiles balance upper and lower
first partial moments, unlike quantiles, which balance tail
probabilities.

\subsection{Basic properties}

The following properties are immediate from
\eqref{eq:bg:expectile-def} or \eqref{eq:bg:expectile-foc}; we collect
them without proof and refer to \textcite[\S2]{NeweyPowell1987} and
\textcite[Sections~2--3]{BelliniKlarMullerRosazzaGianin2014} for
details.

\begin{proposition}
\label{prop:bg:expectile-properties}
Let $X$ and $Y$ be real-valued random variables with $\E |X|, \E |Y|
< \infty$. For each $\tau \in (0, 1)$:
\begin{enumerate}[label=(\roman*)]
\item \textbf{Existence and uniqueness.} $\expectile{\tau}(X)$ exists
      and is uniquely defined.
\item \textbf{Law invariance.} If $X$ and $Y$ have the same
      distribution, then $\expectile{\tau}(X)=\expectile{\tau}(Y)$.
\item \textbf{Mean recovery.} $\expectile{1/2}(X) = \E X$.
\item \textbf{Location equivariance.}
      $\expectile{\tau}(X + c) = \expectile{\tau}(X) + c$ for
      $c \in \R$.
\item \textbf{Positive homogeneity.}
      $\expectile{\tau}(\lambda X) = \lambda\, \expectile{\tau}(X)$
      for $\lambda \ge 0$.
\item \textbf{Sign reflection.}
      $\expectile{\tau}(-X) = -\expectile{1 - \tau}(X)$.
\item \textbf{Monotonicity in $\tau$.} The map
      $\tau \mapsto \expectile{\tau}(X)$ is nondecreasing on
      $(0, 1)$, and is strictly increasing when $X$ is nondegenerate,
      with the boundary limits
      $\expectile{\tau}(X) \to \mathrm{ess}\inf(X)$ as $\tau \to 0$
      and $\expectile{\tau}(X) \to \mathrm{ess}\sup(X)$ as
      $\tau \to 1$ (infinite limits allowed).
\item \textbf{Monotonicity in the argument.} If $X \le Y$ almost
      surely, then $\expectile{\tau}(X) \le \expectile{\tau}(Y)$.
\end{enumerate}
\end{proposition}

For losses, expectiles are translation equivariant, positively
homogeneous and monotone. The remaining coherence issue is
subadditivity.

\subsection{Coherence and elicitability}

In the rest of the thesis $X$ denotes a loss. The coherence literature
is often written for financial positions, so in this subsection only
write $Y$ for a position and define the capital requirement
\[
  \rho_{\tau}(Y) \;=\; \expectile{\tau}(-Y).
\]

\begin{theorem}[\textcite{BelliniKlarMullerRosazzaGianin2014}]
\label{thm:bg:expectile-coherence}
The functional $\rho_{\tau}(Y) = \expectile{\tau}(-Y)$ is a coherent
risk measure if and only if $\tau \ge 1/2$.
\end{theorem}

Recall that a functional is \emph{elicitable} if it is the unique
minimiser of an expected strictly consistent score
\parencite{Gneiting2011}. Expectiles are elicitable on classes with
finite first moment; the asymmetric squared score is strictly
consistent under finite second moments, and the general scoring class
has Bregman-type and Choquet-type representations
\parencite{Gneiting2011,EhmGneitingJordanKrueger2016}. This scoring
structure motivates forecast comparison, but the asymptotic theory below
uses only \eqref{eq:bg:expectile-foc} and the extreme
expectile--quantile equivalence.

\section{Extreme expectile inference: the iid heavy-tailed baseline}
\label{sec:bg:extreme-expectile}

We now consider high expectile levels for a heavy-tailed loss with
$\gamma\in(0,1)$. Two regimes are used throughout:
\begin{itemize}
\item \emph{intermediate-level}: $\tau_{n} \to 1$ with
      $n(1 - \tau_{n}) \to \infty$. There are still many observations
      above $\Quant_{X}(\tau_{n})$, and central limit theorems hold at
      the rate $\sqrt{n(1 - \tau_{n})}$.
\item \emph{extreme-level}: $\tau_{n}^{\prime} \to 1$ with
      $n(1 - \tau_{n}^{\prime}) \to c \in [0, \infty)$. There are at
      most finitely many observations above
      $\Quant_{X}(\tau_{n}^{\prime})$, and direct estimation is
      infeasible; extrapolation is required.
\end{itemize}
The iid results are due to
\textcite{DaouiaGirardStupfler2018,DaouiaGirardStupfler2019,DaouiaGirardStupfler2020};
the plug-in formulation used later is adapted from the time-series
extension of \textcite{DavisonPadoanStupfler2023}.

\subsection{The expectile--quantile asymptotic equivalence}

The single most important fact for extreme expectile inference is
that, in the heavy-tailed regime, the expectile is asymptotically
proportional to the quantile of the same level.

\begin{proposition}[Asymptotic equivalence;
                    \textcite{BelliniKlarMullerRosazzaGianin2014},
                    \textcite{DaouiaGirardStupfler2019}, Corollary~1 with $p=2$]
\label{prop:bg:expectile-quantile}
Suppose $X$ satisfies \Cref{cond:bg:first-order} with index
$\gamma \in (0, 1)$ and $\E X^{-} < \infty$; equivalently, suppose
\Cref{cond:bg:finite-first-moment} holds. Then
\begin{equation}
\label{eq:bg:expectile-quantile-equivalence}
  \frac{\expectile{\tau}(X)}{\Quant_{X}(\tau)}
  \;\xrightarrow{\tau \uparrow 1}\;
  \psifn(\gamma) \;\coloneqq\; \big(\gamma^{-1} - 1\big)^{-\gamma}.
\end{equation}
\end{proposition}

The bridge requires only the finite-mean range $\gamma<1$ and lower-tail
integrability; the stronger $\gamma<1/2$ restriction enters direct LAWS
CLTs, not the quantile-based route. The function $\psifn$ is continuous
on $(0,1)$, with
$\psifn(\gamma) \to 1$ as $\gamma \downarrow 0$ and
$\psifn(\gamma) \to \infty$ as $\gamma \uparrow 1$. It is \emph{not}
monotone: its logarithmic derivative
\begin{equation}
\label{eq:bg:m-function}
  m(\gamma)
  \;\coloneqq\;
  \frac{d}{d\gamma}\log\psifn(\gamma)
  \;=\;
  \frac{1}{1 - \gamma} - \log\big(\gamma^{-1} - 1\big)
\end{equation}
changes sign at $\gamma^{\star}\approx0.2178$, where $\psifn$ reaches
its minimum. This non-monotonicity is relevant later when
$m(\gamma)=0$ makes some weight formulas degenerate.

Under the second-order condition $\mathcal C_{2}(\gamma, \rho, A)$ with
$\gamma \in (0, 1)$ and $\E X^{-}<\infty$, the first-order equivalence
\eqref{eq:bg:expectile-quantile-equivalence} sharpens to the
second-order expansion
\parencite[Proposition~1(i)]{DaouiaGirardStupfler2020}
\begin{equation}
\label{eq:bg:c-gamma-rho}
  \begin{aligned}
  \frac{\expectile{\tau}(X)}{\Quant_{X}(\tau)}
  &=
  \psifn(\gamma)\Big[
      1 + c(\gamma, \rho)\,A\big((1 - \tau)^{-1}\big)
      + \gamma(\gamma^{-1} - 1)^{\gamma}
        \frac{\E X}{\Quant_{X}(\tau)}
  \\
  &\hspace{5.6em}
      + o\big(A((1 - \tau)^{-1})\big)
      + o\big(\Quant_{X}(\tau)^{-1}\big)
    \Big],
  \\
  c(\gamma, \rho)
  &\coloneqq
  \frac{(\gamma^{-1} - 1)^{-\rho}}{1 - \gamma - \rho}
  + \frac{(\gamma^{-1} - 1)^{-\rho} - 1}{\rho}.
  \end{aligned}
\end{equation}
At $\rho=0$ the second term in $c(\gamma,\rho)$ is read by continuity,
so $c(\gamma,0)=m(\gamma)$. The $A$-term is the second-order
expectile--quantile bias used later; the displayed first-moment term is
$O(\Quant_X(\tau)^{-1})$ and is negligible under the moment-rate
condition imposed in the QB CLT below.

\subsection{Two families of estimators at intermediate levels}

Two natural estimators of $\expectile{\tau_{n}}$ at an intermediate
level $\tau_{n}$ arise from the asymmetric-LS definition
\eqref{eq:bg:expectile-def} on the one hand, and from the asymptotic
equivalence \eqref{eq:bg:expectile-quantile-equivalence} on the other.

\paragraph{LAWS direct estimator.}
The \emph{asymmetric-least-squares} (LAWS) estimator replaces the
population minimisation \eqref{eq:bg:expectile-def} with its empirical
analogue:
\begin{equation}
\label{eq:bg:laws}
  \tildexpectile{\tau_{n}}
  \;=\;
  \argmin_{\theta \in \R}\,
  \sum_{i=1}^{n} \eta_{\tau_{n}}(X_{i} - \theta).
\end{equation}
\begin{theorem}[Intermediate LAWS CLT;
                \textcite{DaouiaGirardStupfler2018}, Theorem~2]
\label{thm:bg:laws-iid}
Assume \Cref{cond:bg:first-order} with
$\gamma \in (0,1/2)$, and assume
\Cref{cond:bg:lower-tail-moment}. Let
$\tau_{n} \to 1$ with $n(1 - \tau_{n}) \to \infty$. Then
\[
  \sqrt{n(1 - \tau_{n})}\!
  \left( \frac{\tildexpectile{\tau_{n}}}{\expectile{\tau_{n}}} - 1
  \right)
  \;\todist\;
  \mathcal N\!\left(0,\; \frac{2 \gamma^{3}}{1 - 2\gamma}\right).
\]
\end{theorem}

\paragraph{Quantile-based plug-in estimator.}
A second family substitutes estimators of $\gamma$ and of the
intermediate quantile $\Quant_{X}(\tau_{n})$ into the
asymptotic-equivalence formula
\eqref{eq:bg:expectile-quantile-equivalence}:
\begin{equation}
\label{eq:bg:qb}
  \hatexpectile{\tau_{n}}
  \;=\;
  \psifn\!\big(\hat\gamma_{n}\big)\, \hatQuant(\tau_{n}),
\end{equation}
where $\hatQuant(\tau_{n}) = X_{n - \lfloor n(1 - \tau_{n}) \rfloor : n}$
is the empirical quantile (an intermediate-order statistic) and
$\hat\gamma_{n}$ is a tail-index estimator admitting the required
$\sqrt{n(1 - \tau_{n})}$ limit theory. The natural choice is the Hill
estimator of \eqref{eq:bg:hill} with effective sample size
$k = \lfloor n(1 - \tau_{n})\rfloor$.

\begin{theorem}[Intermediate QB CLT;
                \textcite{DaouiaGirardStupfler2018}, Corollary~2;
                \textcite{DaouiaGirardStupfler2020}, Proposition~1]
\label{thm:bg:qb-iid}
Assume $\mathcal C_{2}(\gamma, \rho, A)$ with
$\gamma \in (0,1)$ and assume $\E X^{-}<\infty$. Let
$\tau_n \to 1$ with $n(1-\tau_n)\to\infty$ and suppose
\[
  \sqrt{n(1-\tau_n)}\, A\big((1-\tau_n)^{-1}\big)\to \lambda\in\R,
  \qquad
  \frac{\sqrt{n(1-\tau_n)}}{\Quant_X(\tau_n)}\to 0 .
\]
With $\hat\gamma_{n}$ the Hill estimator at effective sample size
$k = \lfloor n(1 - \tau_{n}) \rfloor$,
\[
  \sqrt{n(1 - \tau_{n})}\!
  \left(\frac{\hatexpectile{\tau_{n}}}{\expectile{\tau_{n}}} - 1
  \right)
  \;\todist\;
  \mathcal N\!\left(\Big(\frac{m(\gamma)}{1 - \rho}
                          - c(\gamma, \rho)\Big)\lambda,\;
                    \gamma^{2}\big(1 + m(\gamma)^{2}\big)\right),
\]
with $m(\gamma)$ as in \eqref{eq:bg:m-function} and $c(\gamma, \rho)$
the second-order expectile--quantile constant of
\eqref{eq:bg:c-gamma-rho}. The asymptotic bias has two pieces: the
Hill bias $\lambda / (1 - \rho)$ propagated through the logarithmic
derivative $m(\gamma)$ of $\psifn$, and the deterministic
expectile--quantile gap $-c(\gamma, \rho)\,\lambda$. The empirical
quantile $\hatQuant(\tau_{n})$ contributes no asymptotic mean term at
the $\sqrt{n(1-\tau_n)}$ scale when centered at
$\Quant_X(\tau_n)$; the condition
$\sqrt{n(1 - \tau_{n})}\,/\Quant_{X}(\tau_{n}) \to 0$ inherited from
the theorem ensures that the moment-term contribution to the
expectile--quantile gap is negligible.
\end{theorem}

This is the zero-first-moment-bias specialisation of
\textcite[Corollary~2]{DaouiaGirardStupfler2018}, combined with
\textcite[Proposition~1]{DaouiaGirardStupfler2020}. The imposed
$\sqrt{n(1-\tau_n)}/\Quant_X(\tau_n)\to0$ removes the additional
first-moment bias term present in the more general statement and is the
finite-mean QB regime used in the contribution chapters.

\paragraph{Comparison.}
On the direct-LAWS range $\gamma\in(0,1/2)$ neither LAWS nor QB
uniformly dominates: their asymptotic variances cross at
$\gamma\approx0.2623$, and the LAWS variance diverges as
$\gamma\uparrow1/2$. The thesis uses QB because it composes with the
pooled Hill and Weissman quantile primitives of
\textcite{DaouiaPadoanStupfler2021}, not because it is uniformly more
efficient.

\subsection{Weissman extrapolation to very-extreme expectile levels}
\label{sec:bg:weissman-expectile}

For levels $\tau_{n}^{\prime}$ so close to one that
$n(1 - \tau_{n}^{\prime}) \to c \in [0, \infty)$, neither
$\tildexpectile{\tau_{n}^{\prime}}$ nor
$\hatexpectile{\tau_{n}^{\prime}}$ provides a consistent direct
estimator of $\expectile{\tau_n'}$.
Extrapolation from an intermediate level $\tau_{n}$ to the extreme
level $\tau_{n}^{\prime}$ is required.

The asymptotic equivalence
\eqref{eq:bg:expectile-quantile-equivalence} and the first-order
identity $U(t y) / U(t) \to y^{\gamma}$ together yield the heuristic
\[
  \frac{\expectile{\tau_{n}^{\prime}}}{\expectile{\tau_{n}}}
  \;\sim\;
  \frac{\Quant_{X}(\tau_{n}^{\prime})}{\Quant_{X}(\tau_{n})}
  \;\sim\;
  \left(\frac{1 - \tau_{n}^{\prime}}{1 - \tau_{n}}\right)^{\!-\gamma},
\]
which suggests the \emph{Weissman-type} extrapolated estimator
\begin{equation}
\label{eq:bg:weissman-expectile}
  \overline{\xi}^{\,\star}_{\tau_{n}^{\prime}}
  \;=\;
  \left(\frac{1 - \tau_{n}^{\prime}}{1 - \tau_{n}}\right)^{\!-\hat\gamma_{n}}\!
  \overline{\xi}_{\tau_{n}},
\end{equation}
where $\overline{\xi}_{\tau_{n}}$ is any consistent intermediate-level
estimator satisfying a $\sqrt{n(1-\tau_n)}$-CLT, such as
$\tildexpectile{\tau_{n}}$ or $\hatexpectile{\tau_{n}}$ under the
conditions above, and $\hat\gamma_{n}$ is a
$\sqrt{n(1-\tau_n)}$-consistent tail-index estimator. We write
$\tilde{\xi}^{\,\star}$ and $\hat{\xi}^{\,\star}$ for the
LAWS-anchored and QB-anchored extrapolations, respectively.

\begin{theorem}[Extreme-level CLT;
                \textcite{DaouiaGirardStupfler2018}, Corollaries~3--4]
\label{thm:bg:weissman-expectile-iid}
Assume $\mathcal C_{2}(\gamma,\rho,A)$ with
$\gamma\in(0,1)$ and $\rho<0$, and assume $\E X^{-}<\infty$. Let
$\tau_n\to1$ satisfy $n(1-\tau_n)\to\infty$ and let
$\tau_{n}^{\prime} \to 1$ satisfy
$n(1-\tau'_n)\to c\in[0,\infty)$,
$(1 - \tau_{n}^{\prime})/(1 - \tau_{n}) \to 0$ and
$\sqrt{n(1 - \tau_{n})}\,/\log((1 - \tau_{n})/(1 - \tau_{n}^{\prime}))
\to \infty$. Suppose moreover that
\[
  \sqrt{n(1-\tau_n)}\,A((1-\tau_n)^{-1})\to\lambda\in\R,
  \qquad
  \frac{\sqrt{n(1-\tau_n)}}{\Quant_X(\tau_n)}\to 0 .
\]
If $\hat\gamma_n$ is the Hill estimator and
$\overline\xi_{\tau_n}$ is the QB intermediate estimator of
\Cref{thm:bg:qb-iid}, then
\[
  \frac{\sqrt{n(1 - \tau_{n})}}
       {\log\!\big((1 - \tau_{n})/(1 - \tau_{n}^{\prime})\big)}
  \!\left(\frac{\overline{\xi}^{\,\star}_{\tau_{n}^{\prime}}}
                {\expectile{\tau_{n}^{\prime}}} - 1\right)
  \;\todist\;
  \mathcal N\!\left(\frac{\lambda}{1 - \rho},\; \gamma^{2}\right).
\]
The same conclusion holds with $\overline\xi_{\tau_n}$ equal to the
LAWS estimator of \Cref{thm:bg:laws-iid} under the additional
direct-LAWS assumptions $\gamma\in(0,1/2)$ and
\Cref{cond:bg:lower-tail-moment}.
\end{theorem}

The log-rate is the classical Weissman rate. At this scale the anchor
estimator is first-order invisible, so the limit variance is
$\gamma^2$ for both LAWS and QB anchors.

\subsection{The plug-in central limit theorem}
\label{sec:bg:plug-in}

For pooled inference we need a version that treats
$(\hat\gamma_n,\overline\xi_{\tau_n})$ as generic inputs. The following
thesis-level lemma is adapted from
\textcite[Supplement, Proposition~C.6]{DavisonPadoanStupfler2023}; it
uses a generic rate $a_n$ and makes the deterministic extrapolation
error explicit.

\begin{proposition}[Plug-in CLT for extrapolated expectiles;
                    adapted from
                    \textcite{DavisonPadoanStupfler2023}, Prop.~C.6]
\label{prop:bg:plug-in-clt}
Assume $\mathcal C_2(\gamma,\rho,A)$ with
$\gamma\in(0,1)$ and $\rho<0$, and assume $\E X^{-}<\infty$.
Let $a_n\to\infty$, let $\hat\gamma_n$ be an estimator of $\gamma$,
and let $\widehat\xi_{\tau_n}^{\,I}$ be an intermediate-level
estimator of $\expectile{\tau_n}$ satisfying the joint asymptotic
representation
\begin{equation}
\label{eq:bg:joint-input}
  a_n
    \begin{pmatrix}
      \hat\gamma_{n} - \gamma \\
      \widehat\xi_{\tau_n}^{\,I} / \expectile{\tau_{n}} - 1
    \end{pmatrix}
  \;\todist\;
  \begin{pmatrix} \Gamma \\ \Delta \end{pmatrix},
\end{equation}
for some random vector $(\Gamma, \Delta)^{\top}$. Let
$\tau_n,\tau'_n\to1$ with $n(1-\tau_n)\to\infty$,
$n(1-\tau'_n)\to c\in[0,\infty)$,
$(1 - \tau_{n}^{\prime})/(1 - \tau_{n}) \to 0$, put
\[
  \ell_n
  \;\coloneqq\;
  \log\!\left(\frac{1-\tau_n}{1-\tau'_n}\right),
\]
and suppose
\[
  \frac{a_n}{\ell_n}\to\infty,\qquad
  a_n A((1-\tau_n)^{-1})\to \lambda_1\in\R,\qquad
  \frac{a_n}{\Quant_X(\tau_n)}\to \lambda_2\in\R .
\]
Assume moreover that $\Prob(\widehat\xi_{\tau_n}^{\,I}>0)\to1$ and
that the deterministic extrapolation error satisfies
\[
  \frac{a_n}{\ell_n}
  \left\{
    \log\expectile{\tau'_n}
    - \log\expectile{\tau_n}
    - \gamma\ell_n
  \right\}
  \;\longrightarrow\; 0.
\]
Then the Weissman-extrapolated expectile
\[
  \widehat\xi_{\tau'_n}^{\,\star}
  =
  \left(\frac{1-\tau'_n}{1-\tau_n}\right)^{-\hat\gamma_n}
  \widehat\xi_{\tau_n}^{\,I}
\]
satisfies
\[
  \frac{a_n}
       {\log\!\big((1 - \tau_{n})/(1 - \tau_{n}^{\prime})\big)}
  \!\left(\frac{\widehat\xi_{\tau'_n}^{\,\star}}
                {\expectile{\tau_{n}^{\prime}}} - 1\right)
  \;\todist\;
  \Gamma.
\]
\end{proposition}

\begin{proof}[Proof sketch]
The statement is the log-scale Slutsky argument behind
\textcite[Supplement, Proposition~C.6]{DavisonPadoanStupfler2023},
written here as a background tool. The detailed thesis proof will be
restored only if this plug-in route is used in the rebuilt main theorem.
\end{proof}

\begin{remark}[Deterministic ratio control]
\label{rem:bg:plugin-ratio-control}
The deterministic log-ratio condition in \Cref{prop:bg:plug-in-clt} is
the high-level extrapolation-bias control
\[
  \frac{a_n}{\ell_n}
  \left\{
    \log\expectile{\tau'_n}
    - \log\expectile{\tau_n}
    - \gamma\ell_n
  \right\}
  \;\longrightarrow\; 0.
\]
In the iid heavy-tailed setting this control is supplied by the
very-extreme expectile extrapolation argument of
\textcite[Theorem~5]{DaouiaGirardStupfler2020}, invoked in
\textcite[Proposition~C.6]{DavisonPadoanStupfler2023}. It uses the
sequential Weissman expansion along
$y_n=(1-\tau_n)/(1-\tau_n^\prime)\to\infty$ together with
\eqref{eq:bg:c-gamma-rho}.
\end{remark}

\Cref{prop:bg:plug-in-clt} remains useful background, but the rebuilt
argument must verify directly whether its speed mechanism applies to
the pooled estimator in \Cref{eq:pool-ext:working-estimator}.

\section{The optimal pooling framework}
\label{sec:bg:pooling}

We now recall the pooling framework of
\textcite{DaouiaPadoanStupfler2021}. The rebuilt thesis uses this
framework as the source theory for pooled tail-index and
extreme-quantile inference before applying the expectile--quantile
bridge. The framework pools tail-index and
extreme-quantile estimators from $m$ samples, allowing heterogeneous
sizes, second-order properties and cross-sample tail dependence.

\subsection{Setup}

Throughout \Cref{sec:bg:pooling}, $m$ is fixed while
$n=\sum_{j=1}^{m}n_j\to\infty$. Let
$\bm X_i=(X_{i,1},\ldots,X_{i,m})^\top$, $i\ge1$, be iid copies of
$\bm X=(X_1,\ldots,X_m)^\top$. Margin $j$ is observed for
$i=1,\ldots,n_j$ and has continuous right-heavy-tailed distribution
$F_j$, tail quantile function $U_j=(1/\overline F_j)^\leftarrow$ and
extreme-value index $\gamma_j>0$; its order statistics are denoted
$X_{1:n_j,j}\le\cdots\le X_{n_j:n_j,j}$.

Tail inference in sample $j$ uses its top $k_{j} + 1$ order statistics,
where the effective sample size $k_{j} = k_{j}(n)$ satisfies
$k_{j} \to \infty$ and $k_{j} / n_{j} \to 0$ as $n \to \infty$. The
total effective sample size is $k = \sum_{j=1}^{m} k_{j}$. Assume the
asymptotic-proportionality conditions
\begin{equation}
\label{eq:bg:pool-rates}
  \frac{n_{1}}{n_{j}} \;\to\; b_{j} \in (0, \infty),
  \qquad
  \frac{k_{1}}{k_{j}} \;\to\; c_{j} \in (0, \infty),
\end{equation}
with $b_{1} = c_{1} = 1$. The marginal $F_{j}$ satisfies
$\mathcal C_{2}(\gamma_{j}, \rho_{j}, A_{j})$
(\Cref{cond:bg:second-order}). The thesis works under the
\emph{common-tail-index} assumption
\[
  (H_{\gamma}) \qquad
  \gamma_{1} \;=\; \cdots \;=\; \gamma_{m} \;=\; \gamma.
\]
Under this assumption, all common-$\gamma$ vector limits below are
expressed on the aggregate scale $\sqrt{k}$, with the constants $c_j$
translating the local $\sqrt{k_j}$ rates.

The cross-sample asymptotic dependence is captured by a pairwise
\emph{tail-copula} structure
\parencite[\S2.1]{DaouiaPadoanStupfler2021}.

\paragraph{Condition $\mathcal J(\bm R)$ (tail-copula condition).}
For each pair $(j, \ell)$ with $j \ne \ell$, there exists a function
$\Rjl : [0, \infty]^{2} \setminus \{(\infty, \infty)\} \to [0, \infty)$
such that, for every
$(x_{j}, x_{\ell}) \in [0, \infty]^{2} \setminus \{(\infty,\infty)\}$,
\[
  \lim_{s \to \infty}
    s\, \Prob\!\big(\,
      1 - F_{j}(X_{j}) \le x_{j}/s,\;
      1 - F_{\ell}(X_{\ell}) \le x_{\ell}/s\,\big)
  \;=\; \Rjl(x_{j}, x_{\ell}).
\]

The case $\Rjl \equiv 0$ is asymptotic tail independence, implied by the
independent samples used later in the thesis.

\subsection{The pooled Hill estimator}

Each sample contributes a Hill estimator $\hat\gamma_{j}(k_{j})$ as
in \eqref{eq:bg:hill}; we collect these into the random vector
\[
  \hat{\bm\gamma}_{n}
  \;=\;
  \big(\hat\gamma_{1}(k_{1}), \ldots, \hat\gamma_{m}(k_{m})\big)^{\top}.
\]
Given a weight vector $\bm\omega \in \R^{m}$ with $\bm\omega^{\top}
\bm 1 = 1$, the \emph{pooled Hill estimator} is the weighted arithmetic
combination
\begin{equation}
\label{eq:bg:pooled-hill}
  \hat\gamma_{n}(\bm\omega)
  \;\coloneqq\;
  \bm\omega^{\top}\hat{\bm\gamma}_{n}
  \;=\;
  \sum_{j=1}^{m} \omega_{j}\,\hat\gamma_{j}(k_{j}).
\end{equation}
The affine constraint $\bm\omega^\top\bm 1=1$ ensures consistency for the
common $\gamma$ under $(H_\gamma)$.

The fundamental asymptotic result is the joint CLT for the marginal
Hill estimators.

\begin{theorem}[\textcite{DaouiaPadoanStupfler2021}, Theorem~1]
\label{thm:bg:pool-hill}
Assume $\mathcal C_{2}(\gamma_{j}, \rho_{j}, A_{j})$ for every $j$,
the tail-copula condition $\mathcal J(\bm R)$, the
proportionality conditions \eqref{eq:bg:pool-rates}, and
$\sqrt{k_{j}}\, A_{j}(n_{j}/k_{j}) \to \lambda_{j} \in \R$ for every
$j$. Then
\[
  \big(\sqrt{k_{1}}(\hat\gamma_{1}(k_{1}) - \gamma_{1}),
       \ldots,
       \sqrt{k_{m}}(\hat\gamma_{m}(k_{m}) - \gamma_{m})\big)^{\top}
  \;\todist\;
  \mathcal N(\bm B, \bm V),
\]
with $\bm B = \big(\lambda_{j}/(1 - \rho_{j})\big)_{j=1}^{m}$ and
$\bm V$ the symmetric matrix whose entries are
\[
  V_{j, \ell}
  \;=\;
  \begin{cases}
    \gamma_{j}^{2}, & j = \ell, \\[2pt]
    \displaystyle
    \frac{\gamma_{j}\gamma_{\ell}}
         {\max(b_j,b_\ell)\sqrt{c_jc_\ell}}\,
      \Rjl(b_jc_\ell,\, b_\ell c_j),
        & j < \ell.
  \end{cases}
\]
The matrix is symmetric, so $V_{\ell,j}=V_{j,\ell}$ for $\ell>j$.
Suppose, in addition, the common-$\gamma$ condition
$(H_{\gamma})$. Then the same marginal estimators, now viewed on the
common $\sqrt{k}$ scale, satisfy
\[
  \sqrt{k}\,\big(\hat{\bm\gamma}_{n}-\gamma\bm 1\big)
  \;\todist\;
  \mathcal N(\Bc,\Vc),
\]
where
\[
  (\Bc)_j
  =
  \sqrt{c_j\sum_{i=1}^{m} c_i^{-1}}\,
  \frac{\lambda_j}{1-\rho_j}
  \qquad j=1,\ldots,m.
\]
The covariance matrix $\Vc$ is given by
\[
  (\Vc)_{j,\ell}
  =
  \left(\sum_{i = 1}^{m} c_i^{-1}\right)
  \begin{cases}
    \gamma^{2}c_j, & j=\ell, \\[2pt]
    \displaystyle
    \frac{\gamma^{2}}{\max(b_j,b_\ell)}\,
      \Rjl(b_jc_\ell,b_\ell c_j),
      & j<\ell,
  \end{cases}
\]
with $(\Vc)_{\ell,j}=(\Vc)_{j,\ell}$ for $\ell>j$.
Consequently, for any $\bm\omega$ with $\bm\omega^{\top}\bm 1 = 1$,
\[
  \sqrt{k}\,\big(\hat\gamma_{n}(\bm\omega) - \gamma\big)
  \;\todist\;
  \mathcal N\!\big(\bm\omega^{\top}\Bc,\;
                    \bm\omega^{\top}\Vc\bm\omega\big).
\]
\end{theorem}

For independent samples, all off-diagonal entries vanish and $\Vc$
becomes the diagonal matrix
\begin{equation}
\label{eq:bg:V_c-diagonal}
  \Vc
  \;=\;
  \gamma^{2}\!\left(\sum_{i = 1}^{m} c_{i}^{-1}\right)
    \mathrm{diag}(c_{1}, \ldots, c_{m}).
\end{equation}
\Cref{eq:bg:V_c-diagonal} is the covariance form to be used when the
rebuilt main theorem is specialised to independent samples.

\subsection{Variance- and AMSE-optimal weights}

The relevant weights solve quadratic programmes under the affine
constraint $\bm\omega^\top\bm 1=1$; unless stated otherwise, they need
not be nonnegative.

\begin{proposition}[Optimal weights;
                    \textcite{DaouiaPadoanStupfler2021}, Theorem~1;
                    estimated-weight implementation in \S2.2 and Corollary~1]
\label{prop:bg:optimal-weights}
Suppose $\Vc$ is positive definite. Under the common-$\gamma$
condition $(H_{\gamma})$:
\begin{enumerate}[label=(\roman*)]
\item The variance-optimal weights, minimising
      $\bm\omega^{\top}\Vc\bm\omega$ subject to
      $\bm\omega^{\top}\bm 1 = 1$, are
      \[
        \bm\omega^{\mathrm{Var}}
        \;=\;
        \frac{\Vc^{-1}\bm 1}{\bm 1^{\top}\Vc^{-1}\bm 1},
      \]
      with corresponding asymptotic variance
      $\bm\omega^{\mathrm{Var},\top}\Vc\bm\omega^{\mathrm{Var}}
        = (\bm 1^{\top}\Vc^{-1}\bm 1)^{-1}$.
\item The AMSE-optimal weights, minimising
      $(\bm\omega^{\top}\Bc)^{2} + \bm\omega^{\top}\Vc\bm\omega$
      subject to $\bm\omega^{\top}\bm 1 = 1$, are
      \[
        \bm\omega^{\mathrm{AMSE}}
        \;=\;
        \frac{(1 + \Bc^{\top}\Vc^{-1}\Bc)\,\Vc^{-1}\bm 1
              - (\bm 1^{\top}\Vc^{-1}\Bc)\,\Vc^{-1}\Bc}
             {(1 + \Bc^{\top}\Vc^{-1}\Bc)(\bm 1^{\top}\Vc^{-1}\bm 1)
              - (\bm 1^{\top}\Vc^{-1}\Bc)^{2}}.
      \]
\end{enumerate}
\end{proposition}

In the independent-samples case, $\Vc$ is diagonal
(\Cref{eq:bg:V_c-diagonal}) and the variance-optimal weights reduce
to
\begin{equation}
\label{eq:bg:V-opt-distrib}
  \omega_{j}^{\mathrm{Var}}
  \;=\;
  \Bigg(\sum_{i = 1}^{m} c_{i}^{-1}\Bigg)^{\!\!-1}\!
  c_{j}^{-1}
  \;\sim\;
  \frac{k_{j}}{k},
\end{equation}
where the equivalence uses $c_{i}^{-1} = \lim k_{i}/k_{1}$. The
finite-sample distributed version is
\[
  \tilde{\bm\omega}_{n}^{\mathrm{Var}}
  =
  (k_{1}/k,\ldots,k_{m}/k)^{\top},
\]
which is exactly normalised and converges to $\bm\omega^{\mathrm{Var}}$.

Feasible weights replace $\Vc$ and $\Bc$ by consistent plug-ins
\parencite[\S2.2]{DaouiaPadoanStupfler2021}. If
$\hat{\bm\omega}^{\top}\bm 1=1$ exactly and
$\hat{\bm\omega}\toprob\bm\omega$, then
$\hat\gamma_n(\hat{\bm\omega})$ is $\sqrt{k}$-equivalent to
$\hat\gamma_n(\bm\omega)$; the difficult practical step is stable
estimation of $\Vc$ and the bias quantities.

\subsection{Weighted geometric pooling of extreme quantiles}

For estimation of an extreme quantile $q(1 - p)$ with $p = p(n) \to
0$, each sample contributes a Weissman estimator
\[
  \hat q_{j}^{\star}(1 - p \mid k_{j})
  \;=\;
  \left(\frac{k_{j}}{n_{j}\, p}\right)^{\!\hat\gamma_{j}(k_{j})}\!
  X_{n_{j} - k_{j} : n_{j},\,j}.
\]
Because Weissman extrapolation is multiplicative, pooling is geometric
on the original scale, equivalently linear on the log scale. It requires
asymptotic equivalence of the marginal extreme quantiles:
\begin{condition}[Tail homoskedasticity $(\mathcal H)$]
\label{cond:bg:tail-homoskedasticity}
For every $j \ne \ell$, $U_{j}(t) / U_{\ell}(t) \to 1$ as $t \to
\infty$.
\end{condition}
Equivalently, $\Quant_j(1-p)/\Quant_\ell(1-p)\to1$ as $p\downarrow0$.
The pooled extreme-quantile estimator of
\textcite[\S2.3]{DaouiaPadoanStupfler2021} is
\begin{equation}
\label{eq:bg:pooled-weissman}
  \hatWei_{n}(1 - p \mid \bm\omega)
  \;\coloneqq\;
  \prod_{j = 1}^{m}
    \big[\hat q_{j}^{\star}(1 - p \mid k_{j})\big]^{\omega_{j}}.
\end{equation}
Taking logs reduces pooling to weighted sums of the local identities
$\log \hat q_{j}^{\star}(1 - p \mid k_{j})
  = \log(k_{j}/(n_{j} p))\,\hat\gamma_{j}(k_{j})
    + \log X_{n_{j} - k_{j} : n_{j},\,j}$;
so the Hill part is handled by \Cref{thm:bg:pool-hill};
\eqref{eq:bg:pool-rates} gives
$\log(k_j/(n_jp))=\log(k/(np))+O(1)$.

\begin{theorem}[\textcite{DaouiaPadoanStupfler2021}, Theorem~2]
\label{thm:bg:pool-weissman}
Assume the hypotheses of \Cref{thm:bg:pool-hill}, the common-$\gamma$
condition $(H_{\gamma})$, the tail-homoskedasticity
condition $(\mathcal H)$, and, in addition, $\rho_j<0$ for every
$j=1,\ldots,m$. Let $p = p(n) \to 0$ with
$k/(np) \to \infty$ and $\sqrt{k}/\log(k/(np)) \to \infty$. Then for
any $\bm\omega$ with $\bm\omega^{\top}\bm 1 = 1$ and any
$j \in \{1, \ldots, m\}$,
\[
  \frac{\sqrt{k}}{\log\!\big(k/(np)\big)}
  \!\left(\frac{\hatWei_{n}(1 - p \mid \bm\omega)}{\Quant_{j}(1 - p)}
            - 1\right)
  \;\todist\;
  \mathcal N\!\big(\bm\omega^{\top}\Bc,\;
                    \bm\omega^{\top}\Vc\bm\omega\big).
\]
\end{theorem}

The strict $\rho_j<0$ assumption is part of the Weissman theorem. It
also supplies the rate needed to recenter by any marginal
$\Quant_j(1-p)$ under $(\mathcal H)$
\parencite[Supplement, Lemma~B.2]{DaouiaPadoanStupfler2021}: for any
$j,\ell$,
\[
  \log\Quant_j(1-p)-\log\Quant_\ell(1-p)
  =
  O\!\big(\abs{A_j(1/p)}\big)
  +
  O\!\big(\abs{A_\ell(1/p)}\big)
  =
  o(k^{-1/2}),
\]
along the Weissman sequence.

Thus the pooled Weissman and pooled Hill limits share the same mean,
variance and optimal weights up to the log-rate factor. This is the
multi-sample analogue of the tail-index dominance in
\Cref{thm:bg:weissman-expectile-iid,prop:bg:plug-in-clt}.

\subsection{Tail-homogeneity testing and confidence intervals}

The covariance structure $\Vc$ also yields a Wald-type chi-square
test for the common-$\gamma$ hypothesis $(H_{\gamma})$,
sometimes called \emph{tail homogeneity} across samples.

\begin{theorem}[Tail-homogeneity test;
                \textcite{DaouiaPadoanStupfler2021}, Corollary~3]
\label{thm:bg:tail-homogeneity-test}
Assume the hypotheses of \Cref{thm:bg:pool-hill}, the null
$H_{0}: \gamma_{1} = \cdots = \gamma_{m}$, and the bias-free
calibration $\lambda_j=0$ for every $j$. The statistic
\[
  \Lambda_{n}
  \;=\;
  k\,(\hat{\bm\gamma}_{n} - \bar\mu_{n}\bm 1)^{\top}\hatVc^{-1}
       (\hat{\bm\gamma}_{n} - \bar\mu_{n}\bm 1),
  \qquad
  \bar\mu_{n}
  \;=\;
  \frac{\bm 1^{\top}\hatVc^{-1}\hat{\bm\gamma}_{n}}
       {\bm 1^{\top}\hatVc^{-1}\bm 1},
\]
where $\hatVc$ is a consistent estimator of the covariance matrix of
$\sqrt{k}(\hat{\bm\gamma}_n-\gamma\bm 1)$ under $H_0$.
Then $\Lambda_{n} \todist \chi^{2}_{m - 1}$ under $H_{0}$ and
$\Lambda_{n} \toprob \infty$ under fixed alternatives. A test of
asymptotic size $\alpha$ rejects $H_{0}$ when
$\Lambda_{n} > \chi^{2}_{m - 1,\, 1 - \alpha}$.
\end{theorem}

\textcite{DaouiaPadoanStupfler2021} state the bias-free version above.
As a direct Gaussian consequence, rather than as an additional result
stated in that paper, the same GLS calculation also shows that, if the
asymptotic bias is retained rather than removed by undersmoothing or
bias correction, the deviance has the noncentral limit
\[
  \Lambda_n \todist \chi^2_{m-1}(\delta_\gamma),
  \qquad
  \delta_\gamma
  =
  \Bc^\top\Vc^{-1}\Bc
  -
  \frac{(\bm 1^\top\Vc^{-1}\Bc)^2}{\bm 1^\top\Vc^{-1}\bm 1}.
\]
Thus the central chi-square limit holds whenever
$\Bc\in\mathrm{span}\{\bm 1\}$, in particular under
$\lambda_j=0$ for every $j$. Under contiguous alternatives of order
$k^{-1/2}$, the limit is likewise noncentral rather than divergent.

The same Gaussian limits give confidence intervals for the common
$\gamma$ and log-scale intervals for pooled Weissman quantiles; retained
bias must be removed or included in the centering.

\subsection{Distributed inference as a special case}
\label{sec:bg:pool-distrib}

In the independent common-distribution distributed setting, the samples
are iid subdivisions of one common population. Then:
\begin{itemize}
\item the common-$\gamma$ condition $(H_{\gamma})$ and the
      tail-homoskedasticity condition $(\mathcal H)$ are trivially
      satisfied;
\item the tail copula $\Rjl$ between any pair of distinct samples
      vanishes identically (independence implies tail independence);
\item $\Vc$ in \Cref{thm:bg:pool-hill} becomes the diagonal matrix
      \eqref{eq:bg:V_c-diagonal};
\item the variance-optimal weights of \Cref{prop:bg:optimal-weights}
      reduce to $\omega_{j}^{\mathrm{Var}} \propto k_{j}$ as in
      \eqref{eq:bg:V-opt-distrib}.
\end{itemize}

\begin{corollary}[Distributed oracle equivalence;
                  \textcite{DaouiaPadoanStupfler2021},
                  Corollary~6, Theorem~3 and Corollary~8]
\label{cor:bg:pool-distrib-oracle}
Under the distributed-inference assumptions (iid across machines) of
Corollary~5 in \textcite{DaouiaPadoanStupfler2021}, suppose the sample
fractions are asymptotically equal in the stronger sense
$k_{j} / n_{j} = (k/n)(1 + O(k^{-1/2}))$ for every $j$. Then, under the
conditions of their Theorem~3, the variance-optimally-pooled Hill
estimator
$\hat\gamma_{n}(\tilde{\bm\omega}_{n}^{\mathrm{Var}})$ is
$\sqrt{k}$-asymptotically equivalent to the unfeasible benchmark Hill estimator
$\hat\gamma_{n}^{(\mathrm{Hill})}(k)$ built from the entire combined
sample. If, in addition, $\rho<0$ and $p=p(n)$ satisfies
$k/(np)\to\infty$ and $\sqrt{k}/\log(k/(np))\to\infty$, then the
variance-optimally-pooled Weissman estimator
$\hatWei_{n}(1 - p \mid \tilde{\bm\omega}_{n}^{\mathrm{Var}})$ is
asymptotically equivalent to the unfeasible benchmark Weissman estimator
built from the combined sample.
\end{corollary}

\Cref{cor:bg:pool-distrib-oracle} is the efficiency benchmark the
rebuilt pooled-expectile construction should preserve if the
expectile-specific terms in \Cref{eq:pool-ext:central-decomposition}
are negligible on the pooled Weissman scale.
